52/// A source of node group definitions.
53///
54/// Group sources can be backed by various backends: static in-memory data,
55/// shell scripts, configuration files, etc.
56pub trait GroupSource: Send + Sync {
57    /// Resolve a group name to a node pattern string.
58    ///
59    /// Returns a comma-separated list of node names or a node pattern.
60    fn resolve_map(&self, group: &str) -> Result<String>;
61
62    /// Get all nodes from all groups.
63    ///
64    /// Returns a comma-separated list of all nodes.
65    fn resolve_all(&self) -> Result<String> {
66        Err(GroupSourceError::GroupSourceNoUpcall.into())
67    }
68
69    /// List all available group names.
70    fn resolve_list(&self) -> Result<Vec<String>> {
71        Err(GroupSourceError::GroupSourceNoUpcall.into())
72    }
73
74    /// Find all groups that contain a given node.
75    fn resolve_reverse(&self, _node: &str) -> Result<Vec<String>> {
76        Err(GroupSourceError::GroupSourceNoUpcall.into())
77    }
78
79    /// Returns true if this source supports reverse lookups.
80    fn has_reverse(&self) -> bool {
81        false
82    }
83
84    /// Returns the cache time in seconds, if any.
85    fn cache_time(&self) -> Option<f64> {
86        None
87    }
88}

90/// In-memory group source backed by a HashMap.
91///
92/// Supports basic resolve_map, and optionally resolve_all, resolve_list,
93/// and resolve_reverse if provided.
94pub struct StaticGroupSource {
95    map: HashMap<String, String>,
96    all: Option<String>,
97    list: Option<Vec<String>>,
98    reverse: Option<HashMap<String, Vec<String>>>,
99}
100
101impl StaticGroupSource {
102    /// Create a new StaticGroupSource with the given map.
103    pub fn new(map: HashMap<String, String>) -> Self {
104        Self {
105            map,
106            all: None,
107            list: None,
108            reverse: None,
109        }
110    }
111
112    /// Set the all nodes string.
113    pub fn with_all(mut self, all: impl Into<String>) -> Self {
114        self.all = Some(all.into());
115        self
116    }
117
118    /// Set the list of group names.
119    pub fn with_list(mut self, list: Vec<String>) -> Self {
120        self.list = Some(list);
121        self
122    }
123
124    /// Set the reverse lookup map (node -> groups containing it).
125    pub fn with_reverse(mut self, reverse: HashMap<String, Vec<String>>) -> Self {
126        self.reverse = Some(reverse);
127        self
128    }
129}

131impl GroupSource for StaticGroupSource {
132    fn resolve_map(&self, group: &str) -> Result<String> {
133        self.map
134            .get(group)
135            .cloned()
136            .ok_or(GroupSourceError::GroupSourceNoUpcall.into())
137    }
138
139    fn resolve_all(&self) -> Result<String> {
140        self.all
141            .clone()
142            .ok_or(GroupSourceError::GroupSourceNoUpcall.into())
143    }
144
145    fn resolve_list(&self) -> Result<Vec<String>> {
146        self.list
147            .clone()
148            .ok_or(GroupSourceError::GroupSourceNoUpcall.into())
149    }
150
151    fn resolve_reverse(&self, node: &str) -> Result<Vec<String>> {
152        self.reverse
153            .as_ref()
154            .and_then(|r| r.get(node).cloned())
155            .ok_or(GroupSourceError::GroupSourceNoUpcall.into())
156    }
157
158    fn has_reverse(&self) -> bool {
159        self.reverse.is_some()
160    }
161}

163/// Shell-based group source that runs commands for lookups.
164///
165/// Commands can use %s as a placeholder for the group name.
166pub struct UpcallGroupSource {
167    map_upcall: Option<String>,
168    all_upcall: Option<String>,
169    list_upcall: Option<String>,
170    reverse_upcall: Option<String>,
171}
172
173impl UpcallGroupSource {
174    /// Create a new UpcallGroupSource with the given command templates.
175    ///
176    /// Command templates can use %s as a placeholder for the group name.
177    pub fn new(
178        map_upcall: Option<String>,
179        all_upcall: Option<String>,
180        list_upcall: Option<String>,
181        reverse_upcall: Option<String>,
182    ) -> Self {
183        Self {
184            map_upcall,
185            all_upcall,
186            list_upcall,
187            reverse_upcall,
188        }
189    }
190
191    fn run_command(&self, template: &str, arg: &str) -> Result<String> {
192        let command = template.replace("%s", arg);
193        let output = std::process::Command::new("sh")
194            .arg("-c")
195            .arg(&command)
196            .env("GROUP", arg)
197            .output()
198            .map_err(|e| GroupSourceError::GroupSourceQueryFailed(e.to_string()))?;
199
200        if output.status.success() {
201            Ok(String::from_utf8(output.stdout)
202                .map_err(|e| GroupSourceError::GroupSourceQueryFailed(e.to_string()))?
203                .trim()
204                .to_string())
205        } else {
206            let stderr = String::from_utf8_lossy(&output.stderr);
207            Err(GroupSourceError::GroupSourceQueryFailed(stderr.to_string()).into())
208        }
209    }
210}
211
212impl GroupSource for UpcallGroupSource {
213    fn resolve_map(&self, group: &str) -> Result<String> {
214        match self.map_upcall.as_ref() {
215            Some(t) => self.run_command(t, group),
216            None => Err(GroupSourceError::GroupSourceNoUpcall.into()),
217        }
218    }
219
220    fn resolve_all(&self) -> Result<String> {
221        match self.all_upcall.as_ref() {
222            Some(t) => self.run_command(t, ""),
223            None => Err(GroupSourceError::GroupSourceNoUpcall.into()),
224        }
225    }
226
227    fn resolve_list(&self) -> Result<Vec<String>> {
228        match self.list_upcall.as_ref() {
229            Some(t) => {
230                let s = self.run_command(t, "")?;
231                Ok(s.split(|c: char| c == ',' || c.is_whitespace())
232                    .filter(|s| !s.is_empty())
233                    .map(|s| s.to_string())
234                    .collect())
235            }
236            None => Err(GroupSourceError::GroupSourceNoUpcall.into()),
237        }
238    }
239
240    fn resolve_reverse(&self, node: &str) -> Result<Vec<String>> {
241        match self.reverse_upcall.as_ref() {
242            Some(t) => {
243                let s = self.run_command(t, node)?;
244                Ok(s.split(|c: char| c == ',' || c.is_whitespace())
245                    .filter(|s| !s.is_empty())
246                    .map(|s| s.to_string())
247                    .collect())
248            }
249            None => Err(GroupSourceError::GroupSourceNoUpcall.into()),
250        }
251    }
252
253    fn has_reverse(&self) -> bool {
254        self.reverse_upcall.is_some()
255    }
256}

258/// Manages named group sources with a default source.
259///
260/// Resolves group references like `@group` to node sets.
261pub struct GroupResolver {
262    sources: HashMap<String, Box<dyn GroupSource>>,
263    default_source_name: Option<String>,
264    illegal_chars: HashSet<char>,
265}
266
267impl GroupResolver {
268    /// Create a new GroupResolver.
269    pub fn new() -> Self {
270        Self {
271            sources: HashMap::new(),
272            default_source_name: None,
273            illegal_chars: HashSet::new(),
274        }
275    }
276
277    /// Add a named source to the resolver.
278    pub fn add_source(&mut self, name: impl Into<String>, source: Box<dyn GroupSource>) {
279        self.sources.insert(name.into(), source);
280    }
281
282    /// Set the default source name.
283    pub fn set_default(&mut self, name: impl Into<String>) {
284        self.default_source_name = Some(name.into());
285    }
286
287    /// Get the source by name, or the default if name is None.
288    fn get_source(&self, namespace: Option<&str>) -> Result<&dyn GroupSource> {
289        let name = match namespace {
290            Some(n) => n,
291            None => self.default_source_name.as_ref().ok_or(
292                GroupResolverSourceError::SourceNotFound("no default source set".to_string()),
293            )?,
294        };
295
296        self.sources
297            .get(name)
298            .map(|s| s.as_ref())
299            .ok_or(GroupResolverSourceError::SourceNotFound(name.to_string()).into())
300    }
301
302    /// Resolve a group to a list of nodes.
303    ///
304    /// If namespace is Some, uses that source; otherwise uses the default.
305    pub fn group_nodes(&self, group: &str, namespace: Option<&str>) -> Result<Vec<String>> {
306        let source = self.get_source(namespace)?;
307        let pattern = source.resolve_map(group)?;
308        Ok(pattern
309            .split(|c: char| c == ',' || c.is_whitespace())
310            .filter(|s| !s.is_empty())
311            .map(|s| s.to_string())
312            .collect())
313    }
314
315    /// Get all nodes from all groups in the specified namespace.
316    pub fn all_nodes(&self, namespace: Option<&str>) -> Result<Vec<String>> {
317        let source = self.get_source(namespace)?;
318        let all = source.resolve_all()?;
319        Ok(all
320            .split(|c: char| c == ',' || c.is_whitespace())
321            .filter(|s| !s.is_empty())
322            .map(|s| s.to_string())
323            .collect())
324    }
325
326    /// List all group names in the specified namespace.
327    pub fn grouplist(&self, namespace: Option<&str>) -> Result<Vec<String>> {
328        let source = self.get_source(namespace)?;
329        source.resolve_list()
330    }
331
332    /// Find all groups containing a given node.
333    pub fn node_groups(&self, node: &str, namespace: Option<&str>) -> Result<Vec<String>> {
334        let source = self.get_source(namespace)?;
335        source.resolve_reverse(node)
336    }
337
338    /// Check if the namespace supports node group lookups.
339    pub fn has_node_groups(&self, namespace: Option<&str>) -> bool {
340        self.get_source(namespace)
341            .map(|s| s.has_reverse())
342            .unwrap_or(false)
343    }
344
345    /// Set illegal characters for group names.
346    pub fn set_illegal_chars(&mut self, chars: HashSet<char>) {
347        self.illegal_chars = chars;
348    }
349
350    /// Check if a group name contains illegal characters.
351    pub fn is_valid_group_name(&self, name: &str) -> Result<()> {
352        for c in name.chars() {
353            if self.illegal_chars.contains(&c) {
354                return Err(GroupResolverSourceError::SourceError(
355                    GroupSourceError::GroupSourceQueryFailed(format!(
356                        "group name contains illegal character: '{}'",
357                        c
358                    )),
359                )
360                .into());
361            }
362        }
363        Ok(())
364    }
365}
366
367impl Default for GroupResolver {
368    fn default() -> Self {
369        Self::new()
370    }

373/// Loads group sources from INI configuration files.
374///
375/// Supports the groups.conf format with:
376/// - `[Main]` section for groupsdir, confdir, autodir, default
377/// - Other sections as UpcallGroupSource definitions
378pub struct GroupResolverConfig {
379    filenames: Vec<PathBuf>,
380    illegal_chars: HashSet<char>,
381    initialized: bool,
382    resolver: Option<GroupResolver>,
383}
384
385impl GroupResolverConfig {
386    /// Create a new GroupResolverConfig with the given filenames.
387    pub fn new(filenames: Vec<PathBuf>, illegal_chars: HashSet<char>) -> Self {
388        Self {
389            filenames,
390            illegal_chars,
391            initialized: false,
392            resolver: None,
393        }
394    }
395
396    /// Initialize the config if not already done.
397    fn ensure_initialized(&mut self) -> Result<&mut GroupResolver> {
398        if !self.initialized {
399            self.initialize()?;
400            self.initialized = true;
401        }
402        Ok(self.resolver.as_mut().unwrap())
403    }
404
405    /// Initialize the resolver from config files.
406    fn initialize(&mut self) -> Result<()> {
407        let mut resolver = GroupResolver::new();
408        resolver.set_illegal_chars(self.illegal_chars.clone());
409
410        for filename in &self.filenames {
411            if !filename.exists() {
412                return Err(GroupResolverConfigError::FileNotFound(filename.clone()).into());
413            }
414
415            let mut ini = Ini::new();
416            // configparser lowercases keys/sections by default
417            ini.load(filename.to_string_lossy().as_ref())
418                .map_err(|e| GroupResolverConfigError::ParseError(e.to_string()))?;
419
420            // Process [Main] section (lowercased by configparser)
421            if let Some(default) = ini.get("main", "default") {
422                resolver.set_default(default);
423            }
424
425            // Process other sections as UpcallGroupSource
426            for section_name in ini.sections() {
427                if section_name == "main" || section_name == "default" {
428                    continue;
429                }
430
431                let map_upcall = ini.get(&section_name, "map");
432                let all_upcall = ini.get(&section_name, "all");
433                let list_upcall = ini.get(&section_name, "list");
434                let reverse_upcall = ini.get(&section_name, "reverse");
435
436                // Only add source if at least map is defined
437                if map_upcall.is_some() {
438                    let source =
439                        UpcallGroupSource::new(map_upcall, all_upcall, list_upcall, reverse_upcall);
440                    resolver.add_source(section_name.clone(), Box::new(source));
441                }
442            }
443        }
444
445        self.resolver = Some(resolver);
446        Ok(())
447    }
448
449    /// Get the underlying resolver.
450    pub fn resolver(&mut self) -> Result<&mut GroupResolver> {
451        self.ensure_initialized()
452    }
453}
