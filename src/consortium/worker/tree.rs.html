1//! TreeWorker implementation for tree-based command propagation.
2//!
3//! Provides TreeWorker, which distributes command execution across a cluster
4//! topology using gateway nodes and propagation channels. Mirrors
5//! ClusterShell's TreeWorker class.
6//!
7//! The TreeWorker uses:
8//! - [`PropagationTreeRouter`] to determine next-hop gateways
9//! - [`PropagationChannel`] to communicate with gateway processes
10//! - Direct `SshWorker`/`ExecWorker` for leaf nodes reachable without gateways
11

20/// Build a NodeSet from a collection of node name strings.
21fn nodeset_from_strings(names: &[String]) -> String {
22    names.join(",")
23}

25/// Configuration for tree-based execution.
26#[derive(Debug, Clone)]
27pub struct TreeWorkerConfig {
28    /// Maximum concurrent operations per gateway.
29    pub fanout: usize,
30    /// Command timeout per node.
31    pub timeout: Option<Duration>,
32    /// Whether to capture stderr.
33    pub stderr: bool,
34    /// Whether targets are remote (SSH) or local.
35    pub remote: bool,
36    /// Whether this is a copy operation.
37    pub source: Option<String>,
38    /// Destination for copy operations.
39    pub dest: Option<String>,
40    /// Whether this is a reverse copy (remote -> local).
41    pub reverse: bool,
42    /// Whether to preserve file attributes in copy.
43    pub preserve: bool,
44}
45
46impl Default for TreeWorkerConfig {
47    fn default() -> Self {
48        Self {
49            fanout: 64,
50            timeout: None,
51            stderr: false,
52            remote: true,
53            source: None,
54            dest: None,
55            reverse: false,
56            preserve: false,
57        }
58    }
59}

61/// Tracks the state of a gateway connection.
62#[derive(Debug)]
63struct GatewayState {
64    /// The propagation channel for this gateway.
65    channel: PropagationChannel,
66    /// Active target nodes being processed through this gateway.
67    active_targets: HashSet<String>,
68    /// Whether the gateway channel has been started.
69    started: bool,
70}

72/// Result from a remote node received through the tree.
73#[derive(Debug, Clone)]
74pub enum TreeWorkerEvent {
75    /// Node produced stdout output.
76    StdOut {
77        node: String,
78        gateway: String,
79        data: Vec<u8>,
80    },
81    /// Node produced stderr output.
82    StdErr {
83        node: String,
84        gateway: String,
85        data: Vec<u8>,
86    },
87    /// Node completed with a return code.
88    Close {
89        node: String,
90        gateway: String,
91        rc: i32,
92    },
93    /// Node timed out.
94    Timeout { node: String, gateway: String },
95    /// Routing event (reroute, unreachable, etc.)
96    Routing {
97        event: String,
98        gateway: String,
99        targets: String,
100    },
101}

103/// Worker that distributes command execution across a cluster topology.
104///
105/// TreeWorker uses [`PropagationTreeRouter`] to determine routing through
106/// gateway nodes, and [`PropagationChannel`] to communicate with gateways.
107/// Nodes directly reachable (same hop as root) are executed via direct
108/// workers (SSH or local).
109///
110/// Mirrors ClusterShell's `TreeWorker` class.
111pub struct TreeWorker {
112    /// Target nodes for this worker.
113    nodes: NodeSet,
114    /// Command to execute (mutually exclusive with source).
115    command: Option<String>,
116    /// Worker configuration.
117    config: TreeWorkerConfig,
118    /// Current worker state.
119    state: WorkerState,
120    /// Router for determining next-hop gateways.
121    router: Option<PropagationTreeRouter>,
122    /// Gateway states indexed by gateway name.
123    gateways: HashMap<String, GatewayState>,
124    /// Return codes collected per node.
125    retcodes: HashMap<String, i32>,
126    /// Events collected from remote nodes.
127    events: Vec<TreeWorkerEvent>,
128    /// Count of closed/completed targets.
129    close_count: usize,
130    /// Total target count (for completion tracking).
131    target_count: usize,
132    /// Whether any node has timed out.
133    has_timeout: bool,
134    /// Unique worker ID for channel communication.
135    worker_id: u64,
136    /// Nodes that timed out.
137    timeout_nodes: HashSet<String>,
138}

240    /// Distribute target nodes across next-hop gateways.
241    ///
242    /// Returns a list of (gateway, targets) pairs. When gateway == targets,
243    /// the nodes are directly reachable (no gateway needed).
244    fn distribute(&self) -> Result<Vec<(NodeSet, NodeSet)>, WorkerError> {
245        let router = self.router.as_ref().ok_or_else(|| {
246            WorkerError::General("router not set; call set_router() or set_topology()".to_string())
247        })?;
248
249        let mut distribution: HashMap<String, NodeSet> = HashMap::new();
250
251        for (gw, dstset) in router.dispatch(&self.nodes) {
252            let gw_str = gw.to_string();
253            distribution
254                .entry(gw_str)
255                .and_modify(|ns| ns.update(&dstset))
256                .or_insert(dstset);
257        }
258
259        Ok(distribution
260            .into_iter()
261            .map(|(k, v)| {
262                let gw = NodeSet::parse(&k).unwrap_or_default();
263                (gw, v)
264            })
265            .collect())
266    }

268    /// Start the tree worker — distribute work across gateways.
269    ///
270    /// This sets up propagation channels for gateway nodes and prepares
271    /// direct execution for leaf nodes.
272    pub fn start(&mut self) -> Result<Vec<(NodeSet, NodeSet)>, WorkerError> {
273        if self.state != WorkerState::Pending {
274            return Err(WorkerError::General(
275                "worker already started or aborted".to_string(),
276            ));
277        }
278
279        self.state = WorkerState::Running;
280
281        let next_hops = self.distribute()?;
282
283        for (gw, targets) in &next_hops {
284            let target_count = targets.len();
285            self.target_count += target_count;
286
287            let gw_str = gw.to_string();
288            let targets_str = targets.to_string();
289
290            // Check if gateway == targets (direct execution, no gateway needed)
291            if gw_str == targets_str {
292                // Direct targets — caller should spawn SSH/local workers for these
293                continue;
294            }
295
296            // Gateway-mediated execution: set up propagation channel
297            let channel = PropagationChannel::new(&gw_str);
298            let gw_state = GatewayState {
299                channel,
300                active_targets: targets.iter().collect(),
301                started: false,
302            };
303            self.gateways.insert(gw_str, gw_state);
304        }
305
306        Ok(next_hops)
307    }

309    /// Start a gateway's propagation channel with topology data.
310    ///
311    /// # Arguments
312    /// * `gateway` - The gateway node name.
313    /// * `topology_bytes` - Serialized topology data to send to gateway.
314    pub fn start_gateway(
315        &mut self,
316        gateway: &str,
317        topology_bytes: &[u8],
318    ) -> Result<(), WorkerError> {
319        let gw_state = self
320            .gateways
321            .get_mut(gateway)
322            .ok_or_else(|| WorkerError::General(format!("unknown gateway: {}", gateway)))?;
323
324        if gw_state.started {
325            return Err(WorkerError::General(format!(
326                "gateway {} already started",
327                gateway
328            )));
329        }
330
331        gw_state.channel.start(topology_bytes);
332        gw_state.started = true;
333
334        // Send the shell command if this is a command execution
335        if let Some(ref cmd) = self.command {
336            let targets: Vec<String> = gw_state.active_targets.iter().cloned().collect();
337            let targets_str = nodeset_from_strings(&targets);
338            gw_state
339                .channel
340                .shell(&targets_str, cmd.as_bytes(), self.worker_id);
341        }
342
343        Ok(())
344    }

346    /// Process a propagation result from a gateway.
347    ///
348    /// Call this when the gateway channel receives results.
349    /// Returns true if the worker is now complete.
350    pub fn process_result(
351        &mut self,
352        gateway: &str,
353        result: PropagationResult,
354    ) -> Result<bool, WorkerError> {
355        match result {
356            PropagationResult::StdOut { ref node, ref data } => {
357                self.events.push(TreeWorkerEvent::StdOut {
358                    node: node.clone(),
359                    gateway: gateway.to_string(),
360                    data: data.clone(),
361                });
362            }
363            PropagationResult::StdErr { ref node, ref data } => {
364                self.events.push(TreeWorkerEvent::StdErr {
365                    node: node.clone(),
366                    gateway: gateway.to_string(),
367                    data: data.clone(),
368                });
369            }
370            PropagationResult::Retcode { ref node, rc } => {
371                self.retcodes.insert(node.clone(), rc);
372                self.close_count += 1;
373
374                // Remove from active targets
375                if let Some(gw_state) = self.gateways.get_mut(gateway) {
376                    gw_state.active_targets.remove(node);
377                }
378
379                self.events.push(TreeWorkerEvent::Close {
380                    node: node.clone(),
381                    gateway: gateway.to_string(),
382                    rc,
383                });
384            }
385            PropagationResult::Timeout { ref node } => {
386                self.timeout_nodes.insert(node.clone());
387                self.has_timeout = true;
388                self.close_count += 1;
389
390                if let Some(gw_state) = self.gateways.get_mut(gateway) {
391                    gw_state.active_targets.remove(node);
392                }
393
394                self.events.push(TreeWorkerEvent::Timeout {
395                    node: node.clone(),
396                    gateway: gateway.to_string(),
397                });
398            }
399            PropagationResult::Routing {
400                ref event,
401                ref gateway,
402                ref targets,
403            } => {
404                self.events.push(TreeWorkerEvent::Routing {
405                    event: event.clone(),
406                    gateway: gateway.clone(),
407                    targets: targets.clone(),
408                });
409            }
410        }
411
412        // Check if we're done
413        let done = self.close_count >= self.target_count && self.target_count > 0;
414        if done {
415            self.state = WorkerState::Done;
416        }
417
418        // Check if gateway has no more active targets
419        self.check_gateway_fini(gateway);
420
421        Ok(done)
422    }

424    /// Record completion of a direct (non-gateway) node.
425    ///
426    /// Called when a direct SSH/local worker reports node completion.
427    pub fn record_direct_close(&mut self, node: &str, rc: i32) {
428        self.retcodes.insert(node.to_string(), rc);
429        self.close_count += 1;
430
431        if self.close_count >= self.target_count && self.target_count > 0 {
432            self.state = WorkerState::Done;
433        }
434    }

449        if let Some(gw_state) = self.gateways.get(gateway) {
450            if gw_state.active_targets.is_empty() {
451                // Gateway is done — could trigger cleanup
452                // The actual channel close is handled by the caller
453            }
454        }

457    /// Write data to all active gateway channels.
458    pub fn write_remote(&mut self, buf: &[u8]) {
459        let worker_id = self.worker_id;
460        for (_, gw_state) in self.gateways.iter_mut() {
461            if !gw_state.active_targets.is_empty() {
462                let targets: Vec<String> = gw_state.active_targets.iter().cloned().collect();
463                let targets_str = nodeset_from_strings(&targets);
464                gw_state.channel.write(&targets_str, buf, worker_id);
465            }
466        }
467    }

481    /// Abort the tree worker.
482    ///
483    /// Marks all remaining gateway targets as failed with EX_PROTOCOL (76).
484    pub fn abort(&mut self) {
485        if self.is_done() {
486            return;
487        }
488
489        // Close all gateway channels
490        for (gateway, gw_state) in self.gateways.iter_mut() {
491            let remaining: Vec<String> = gw_state.active_targets.drain().collect();
492            for node in remaining {
493                self.retcodes.insert(node.clone(), 76); // EX_PROTOCOL
494                self.close_count += 1;
495                self.events.push(TreeWorkerEvent::Close {
496                    node,
497                    gateway: gateway.clone(),
498                    rc: 76,
499                });
500            }
501        }
502
503        self.state = WorkerState::Aborted;
504    }
