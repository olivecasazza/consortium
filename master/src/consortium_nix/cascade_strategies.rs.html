1//! Cost-aware cascade strategies built on petgraph.
2//!
3//! [`Log2FanOut`](crate::cascade::Log2FanOut) ignores the network — it
4//! pairs sources to targets in id order. That's correct under uniform
5//! topology but pessimal when bandwidth varies: the round wall-time is
6//! the slowest edge in the round, so a single (fast-source, slow-target)
7//! pairing stalls the whole round.
8//!
9//! These strategies pick edges *with* knowledge of the network:
10//!
11//! - [`MaxBottleneckSpanning`]: per round, build a bipartite graph of
12//!   (sources × targets) weighted by `bandwidth(src, tgt)`. Greedily
13//!   pick the highest-bandwidth edges until either every source is
14//!   used once or no targets remain. Maximizes the *slowest* edge in
15//!   the round (since round time is gated by it).
16//!
17//! - [`SteinerGreedy`]: same idea but doesn't constrain each source to
18//!   one edge per round — keeps picking the next highest-bandwidth
19//!   (src-with-closure → target-without-closure) edge until no more
20//!   progress. Fires more edges per round; trades off latency vs
21//!   bandwidth contention on the source side.
22
23use std::collections::HashSet;
24
25use crate::cascade::{CascadePlan, CascadeState, CascadeStrategy, NetworkProfile, NodeId};
26
27// Default bandwidth used when an edge has no entry in NetworkProfile.
28// 100 MB/s is a reasonable LAN baseline — strategies with no bandwidth
29// data degenerate to the Log2FanOut ordering.
30const DEFAULT_BW_BYTES_SEC: u64 = 100 * 1024 * 1024;
31
32// ============================================================================
33// MaxBottleneckSpanning
34// ============================================================================
35
36/// Each round: pick edges that maximize the slowest edge in the round.
37///
38/// Algorithm: enumerate all candidate `(src, tgt, bandwidth(src,tgt))`
39/// triples, sort descending by bandwidth, greedily pick edges where
40/// neither src nor tgt has been used yet. Each source serves at most
41/// one target per round, so the round's wall-time is gated by the
42/// slowest assigned edge — and the greedy max-weight pick *maximizes*
43/// that bottleneck.
44///
45/// This is the *max-weight bipartite matching* the cascade wants per
46/// round. We use greedy rather than Hungarian/Hopcroft-Karp because
47/// (a) greedy is within a 2-approximation of optimal, (b) per-round
48/// graphs are small (matching size = O(N) where N is the cascade
49/// size), and (c) each round's matching cost amortizes across log(N)
50/// rounds — so even O(N²) greedy is cheap relative to the actual
51/// `nix copy` work it schedules.
52///
53/// Note: this is NOT a spanning tree algorithm despite the name — it's
54/// a max-weight matching applied iteratively per round. The
55/// *aggregate* deployment topology that emerges across all rounds IS
56/// a spanning tree (each non-seeded node has exactly one parent),
57/// implicitly built up round-by-round.
58pub struct MaxBottleneckSpanning;
59
60impl CascadeStrategy for MaxBottleneckSpanning {
61    fn name(&self) -> &'static str {
62        "max-bottleneck-spanning"
63    }
64
65    fn next_round(&self, state: &CascadeState, net: &NetworkProfile) -> CascadePlan {
66        let sources: Vec<NodeId> = state
67            .nodes
68            .iter()
69            .filter(|n| state.has_closure.contains(&n.id))
70            .filter(|n| !state.failed_nodes.contains(&n.id))
71            .map(|n| n.id)
72            .collect();
73        let targets: Vec<NodeId> = state
74            .nodes
75            .iter()
76            .filter(|n| !state.has_closure.contains(&n.id))
77            .filter(|n| !state.failed_nodes.contains(&n.id))
78            .map(|n| n.id)
79            .collect();
80
81        if sources.is_empty() || targets.is_empty() {
82            return CascadePlan::default();
83        }
84
85        // Enumerate all candidate edges, sort descending by bandwidth.
86        let mut candidates: Vec<(u64, NodeId, NodeId)> = Vec::new();
87        for &src in &sources {
88            for &tgt in &targets {
89                if net.is_partitioned(src, tgt) {
90                    continue;
91                }
92                if state.attempted.contains(&(src, tgt)) {
93                    continue;
94                }
95                let bw = net.bandwidth_of(src, tgt, DEFAULT_BW_BYTES_SEC);
96                candidates.push((bw, src, tgt));
97            }
98        }
99        candidates.sort_by(|a, b| b.0.cmp(&a.0));
100
101        // Greedy max-weight matching: each src and tgt used at most once.
102        let mut used_src: HashSet<NodeId> = HashSet::new();
103        let mut used_tgt: HashSet<NodeId> = HashSet::new();
104        let mut assignments = Vec::new();
105        for (_bw, src, tgt) in candidates {
106            if used_src.contains(&src) || used_tgt.contains(&tgt) {
107                continue;
108            }
109            assignments.push((src, tgt));
110            used_src.insert(src);
111            used_tgt.insert(tgt);
112        }
113
114        CascadePlan {
115            round: state.round,
116            assignments,
117        }
118    }
119}

121// ============================================================================
122// SteinerGreedy
123// ============================================================================
124
125/// Greedy max-bandwidth assignment without the one-edge-per-source cap.
126///
127/// At each round, pick the highest-bandwidth (source, target) edge
128/// where the source has the closure and the target doesn't. Repeat
129/// until no edges remain that match those criteria.
130///
131/// This fires *more edges per round* than `MaxBottleneckSpanning` (a
132/// fast source can serve multiple targets in one round) but is gated
133/// by the source's outbound bandwidth in practice. Use when network
134/// is heavily skewed and a few sources dominate capacity.
135pub struct SteinerGreedy;
136
137impl CascadeStrategy for SteinerGreedy {
138    fn name(&self) -> &'static str {
139        "steiner-greedy"
140    }
141
142    fn next_round(&self, state: &CascadeState, net: &NetworkProfile) -> CascadePlan {
143        let sources: Vec<NodeId> = state
144            .nodes
145            .iter()
146            .filter(|n| state.has_closure.contains(&n.id))
147            .filter(|n| !state.failed_nodes.contains(&n.id))
148            .map(|n| n.id)
149            .collect();
150        let targets: Vec<NodeId> = state
151            .nodes
152            .iter()
153            .filter(|n| !state.has_closure.contains(&n.id))
154            .filter(|n| !state.failed_nodes.contains(&n.id))
155            .map(|n| n.id)
156            .collect();
157
158        if sources.is_empty() || targets.is_empty() {
159            return CascadePlan::default();
160        }
161
162        // Build all candidate (src, tgt, bw) triples, sort descending by bw.
163        let mut candidates: Vec<(u64, NodeId, NodeId)> = Vec::new();
164        for &src in &sources {
165            for &tgt in &targets {
166                if net.is_partitioned(src, tgt) {
167                    continue;
168                }
169                if state.attempted.contains(&(src, tgt)) {
170                    continue;
171                }
172                let bw = net.bandwidth_of(src, tgt, DEFAULT_BW_BYTES_SEC);
173                candidates.push((bw, src, tgt));
174            }
175        }
176        candidates.sort_by(|a, b| b.0.cmp(&a.0));
177
178        // Greedy pick: each src may serve N targets, each tgt receives once.
179        let mut used_tgt: HashSet<NodeId> = HashSet::new();
180        let mut assignments = Vec::new();
181        for (_bw, src, tgt) in candidates {
182            if used_tgt.contains(&tgt) {
183                continue;
184            }
185            assignments.push((src, tgt));
186            used_tgt.insert(tgt);
187        }
188
189        CascadePlan {
190            round: state.round,
191            assignments,
192        }
193    }
194}

200// ============================================================================
201// LevelTreeFanOut — pre-shaped F-ary tree, level-synchronized
202// ============================================================================
203
204/// Pre-shapes the deployment as a balanced F-ary tree using BFS / heap-
205/// style id assignment: parent of node `i` (i >= 1) is `(i-1) / fanout`.
206/// Each round populates exactly one tree level.
207///
208/// Round 0: seed → its `fanout` direct children (level 1)
209/// Round 1: every level-1 node → its `fanout` children (level 2; F²
210///   parallel deploys)
211/// Round 2: F³ parallel
212/// ...
213///
214/// This matches what users expect from `nh`/`nom`-style visualization:
215/// the tree's "level k" is exactly "round k", so renderers show all of
216/// L1 spinning together, then complete, then all of L2 spinning, etc.
217/// In contrast `MaxBottleneckSpanning` greedy-picks per round, which
218/// produces a heavily left-skewed tree where late nodes appear at
219/// surprising depths.
220///
221/// Use this when:
222/// - The deployment topology is known up-front (typical for nix push
223///   to a fixed fleet)
224/// - Visual tree-shape predictability matters more than per-round
225///   bandwidth optimization
226/// - You want the "L1 deploys to all children at once" semantic
227pub struct LevelTreeFanOut {
228    pub fanout: u32,
229}
230
231impl LevelTreeFanOut {
232    pub fn new(fanout: u32) -> Self {
233        assert!(fanout >= 1, "LevelTreeFanOut: fanout must be >= 1");
234        Self { fanout }
235    }
236
237    /// Children of `id` in the heap-style F-ary tree.
238    ///
239    /// Currently unused by `next_round` (which iterates by target and
240    /// walks UP via `alive_ancestor`), but kept for tests + future
241    /// use cases that need to enumerate the planned tree-shape.
242    #[allow(dead_code)]
243    pub(crate) fn tree_children(&self, id: NodeId, n_nodes: u32) -> Vec<NodeId> {
244        let base = id.0.checked_mul(self.fanout);
245        let Some(base) = base else { return Vec::new() };
246        (1..=self.fanout)
247            .filter_map(|k| {
248                let child = base.checked_add(k)?;
249                if child < n_nodes {
250                    Some(NodeId(child))
251                } else {
252                    None
253                }
254            })
255            .collect()
256    }
257
258    /// Heap-style parent: parent of `i` (i >= 1) is `(i-1) / fanout`.
259    /// Returns `None` for the root (id 0) or invalid math.
260    fn tree_parent(&self, id: NodeId) -> Option<NodeId> {
261        if id.0 == 0 {
262            return None;
263        }
264        Some(NodeId((id.0 - 1) / self.fanout))
265    }
266
267    /// Walk up the heap from `id` looking for an ancestor that has the
268    /// closure AND is not in the failed set. Returns:
269    ///
270    /// - `Some(parent)` when the tree-parent is alive with closure
271    ///   (the canonical level-by-level case)
272    /// - `Some(grandparent)` when the parent is in `failed_nodes` and
273    ///   the grandparent is alive (orphan re-routing)
274    /// - `None` when the parent is just "not ready yet" (no closure
275    ///   but also not failed — wait for them)
276    /// - `None` for the root (id 0, no parent)
277    ///
278    /// Critical: only walks PAST a parent that's in `failed_nodes`.
279    /// A parent that's just not-yet-served is waited on, not skipped
280    /// — otherwise round 0 would short-circuit every target directly
281    /// to the seed and break the level-by-level reveal.
282    fn alive_ancestor<'a>(&self, id: NodeId, state: &CascadeState<'a>) -> Option<NodeId> {
283        let mut cur = id;
284        while let Some(parent) = self.tree_parent(cur) {
285            if state.has_closure.contains(&parent) && !state.failed_nodes.contains(&parent) {
286                return Some(parent);
287            }
288            // Parent isn't usable yet. If it's not failed either,
289            // wait for next round — don't skip ahead.
290            if !state.failed_nodes.contains(&parent) {
291                return None;
292            }
293            // Parent is permanently failed — try grandparent.
294            cur = parent;
295        }
296        None
297    }
298}
299
300impl CascadeStrategy for LevelTreeFanOut {
301    fn name(&self) -> &'static str {
302        "level-tree"
303    }
304
305    fn next_round(&self, state: &CascadeState, net: &NetworkProfile) -> CascadePlan {
306        let n_nodes = state.nodes.len() as u32;
307        let mut assignments = Vec::new();
308        let mut used_targets: HashSet<NodeId> = HashSet::new();
309
310        // Iterate over every TARGET that doesn't yet have the closure.
311        // For each, find the best alive source: prefer the heap tree-
312        // parent (matches the canonical level-by-level shape), but if
313        // that parent is dead/failed, walk UP the heap to find an
314        // alive ancestor (the orphan re-routing mechanism). This keeps
315        // failures localized — one failed node only stalls its OWN
316        // serving attempts, not its entire subtree.
317        // Pre-compute the set of "alive sources" (any node with the
318        // closure that isn't in failed_nodes) for fallback re-routing
319        // when the heap ancestor chain is exhausted.
320        let mut alive_sources: Vec<NodeId> = state
321            .nodes
322            .iter()
323            .filter(|n| state.has_closure.contains(&n.id))
324            .filter(|n| !state.failed_nodes.contains(&n.id))
325            .map(|n| n.id)
326            .collect();
327        alive_sources.sort();
328
329        for tgt_id in 0..n_nodes {
330            let tgt = NodeId(tgt_id);
331            if state.has_closure.contains(&tgt) {
332                continue;
333            }
334            if state.failed_nodes.contains(&tgt) {
335                continue;
336            }
337            if used_targets.contains(&tgt) {
338                continue;
339            }
340
341            // Try the heap ancestor chain first (preserves the
342            // canonical level-by-level tree shape).
343            //
344            // We do NOT filter by `state.attempted` for transient
345            // failures — that would block retries. After (anc, tgt)
346            // failed in a previous round (with a Copy/transient
347            // error), the same edge may succeed on the next attempt.
348            // Permanent failures get filtered via `state.failed_nodes`
349            // (populated by the coordinator on non-transient errors).
350            let mut chosen: Option<NodeId> = None;
351            let heap_anc = self.alive_ancestor(tgt, state);
352            if let Some(anc) = heap_anc {
353                if !net.is_partitioned(anc, tgt) {
354                    chosen = Some(anc);
355                }
356            }
357
358            // Fallback: if heap path unusable (no alive ancestor or
359            // edge partitioned), pick ANY alive source. Skip the
360            // wait-not-ready case (heap_anc.is_none()) — those need
361            // to wait for next round, not short-circuit to root.
362            if chosen.is_none() && heap_anc.is_some() {
363                for &src in &alive_sources {
364                    if net.is_partitioned(src, tgt) {
365                        continue;
366                    }
367                    chosen = Some(src);
368                    break;
369                }
370            }
371
372            let Some(src) = chosen else { continue };
373            assignments.push((src, tgt));
374            used_targets.insert(tgt);
375        }
376
377        CascadePlan {
378            round: state.round,
379            assignments,
380        }
381    }
382}

393    /// Test executor: edge duration = closure_size / bandwidth.
394    /// Used so we can compare strategies on total wall-time.
395    struct BandwidthSimExecutor {
396        closure_bytes: u64,
397        default_bw: u64,

400    impl RoundExecutor for BandwidthSimExecutor {
401        fn dispatch(
402            &self,
403            _nodes: &[CascadeNode],
404            edges: &[(NodeId, NodeId)],
405            net: &NetworkProfile,
406        ) -> HashMap<(NodeId, NodeId), Result<Duration, crate::cascade::CascadeError>> {
407            edges
408                .iter()
409                .map(|(src, tgt)| {
410                    let bw = net.bandwidth_of(*src, *tgt, self.default_bw);
411                    let secs = self.closure_bytes as f64 / bw as f64;
412                    let dur =
413                        Duration::from_secs_f64(secs) + net.latency_of(*src, *tgt, Duration::ZERO);
414                    ((*src, *tgt), Ok(dur))
415                })
416                .collect()
417        }
418    }

428    }

457                    };
458                    ((*src, *tgt), outcome)

461        }

465    fn level_tree_walks_up_to_alive_ancestor_for_orphans() {
466        // 15-node binary tree:
467        //       0
468        //     1   2
469        //    3 4 5 6
470        //   7 8 9 10 11 12 13 14
471        //
472        // Kill node 1 (left subtree's whole intermediate level): its
473        // pre-assigned children (3, 4) become orphans. The walk-up
474        // should cause node 0 (root) to adopt them in a later round.
475        // Without re-routing, 7..10 (grandchildren) would never get
476        // served because their tree-parents 3, 4 never receive the
477        // closure.
478        //
479        // With re-routing: round 0 plans (0→1) (0→2). Round 0 outcomes:
480        // (0→1) FAILS (1 is dead), (0→2) succeeds.
481        // Round 1: 1 is failed_node, so heap-walk-up finds 0 alive.
482        // Plan: (0→3), (0→4) (re-routed orphans), (2→5), (2→6).
483        // Round 2: (0→7..14)? No — heap-walk-up for 7 finds parent 3
484        // also failed (never got closure → in_progress check). Walk
485        // up to 1: also failed. Walk to 0: alive → plan (0→7).
486        //
487        // Test: verify >= 13 nodes converge (everyone except 1 + maybe
488        // 7..10 if they couldn't reach root). Actually with re-routing
489        // 0 can serve all of them eventually, so converged should be
490        // 14 (n_nodes - killed_node).
491        let nodes = make_nodes(15);
492        let mut seeded = std::collections::HashSet::new();
493        seeded.insert(NodeId(0));
494        let mut dead = HashSet::new();
495        dead.insert(NodeId(1));
496        let exec = TargetFailExecutor { dead };
497        let result = run_cascade(
498            nodes,
499            seeded,
500            NetworkProfile::default(),
501            &LevelTreeFanOut::new(2),
502            &exec,
503            32,
504            None,
505        );
506        // Node 1 + maybe its descendants 3, 4, 7, 8, 9, 10 are at risk.
507        // With orphan re-routing, only node 1 itself should fail.
508        // Tightened: assert exactly 14 converge (only n1 fails).
509        assert_eq!(
510            result.converged.len(),
511            14,
512            "orphan re-routing should serve every node except the dead one; got {} converged (failed_nodes={:?})",
513            result.converged.len(),
514            result.failed.as_ref().map(|e| e.affected_nodes()),
515        );
516        assert!(!result.is_success(), "n1 should still appear as failed");
517    }

586    /// Skewed network: half the edges are slow (1 MB/s), half are fast (1 GB/s).
587    fn skewed_network(n: u32) -> NetworkProfile {
588        let mut net = NetworkProfile::default();
589        for src in 0..n {
590            for tgt in 0..n {
591                if src == tgt {
592                    continue;
593                }
594                let bw = if (src + tgt) % 2 == 0 {
595                    1024 * 1024 // 1 MB/s slow
596                } else {
597                    1024 * 1024 * 1024 // 1 GB/s fast
598                };
599                net.bandwidth.insert((NodeId(src), NodeId(tgt)), bw);
600            }
601        }
602        net
603    }

631        // 15 nodes, fanout=2, 1 seed → perfect binary heap layout:
