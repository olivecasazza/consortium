1//! Glue between the cascade primitive and the per-host fleet deploy.
2//!
3//! Production deploys build per-host toplevels (`build:hp01`, `build:hp02`,
4//! ...). Each toplevel typically shares a large subgraph with the others
5//! — same nixpkgs base, same kernel, etc. — but the top-of-graph paths
6//! differ per host config.
7//!
8//! The cascade primitive distributes ONE store path to MANY hosts. To
9//! drive it from a heterogeneous fleet, we group targets by their built
10//! toplevel and run one cascade per group:
11//!
12//! - Homogeneous fleet (e.g. `mm01-mm05` all on the same Mac Mini config)
13//!   → 1 cascade, log-N fan-out, big win.
14//! - Heterogeneous fleet (every host different) → N cascades of size 1
15//!   each, which is just direct copy. Same as today's behavior — no
16//!   regression.
17//! - Realistic mid-case (3 hp + 5 mm in one deploy, with 2 unique
18//!   toplevels) → 2 cascades, both fan out within their group.
19//!
20//! Each cascade's seed is the host running cast — typically the dev box,
21//! which already has the closure built locally. That maps cleanly to
22//! `NixCopyExecutor` whose seed-edge runs `nix copy` LOCALLY (no SSH wrap).
23

33/// Per-host input to the grouped cascade copy.
34#[derive(Debug, Clone)]
35pub struct CascadeCopyTarget {
36    /// Fleet name, e.g. "hp01".
37    pub host_name: String,
38    /// SSH addr, e.g. "root@hp01" or "root@192.168.1.121".
39    pub ssh_addr: String,
40    /// Toplevel store path produced by the build phase.
41    pub toplevel_path: String,
42}
43
44/// Result of one grouped-cascade copy run.
45#[derive(Debug, Default)]
46pub struct CascadeCopyResult {
47    /// Hosts whose toplevels successfully reached them.
48    pub copied: Vec<String>,
49    /// Per-host failure reason. Includes both transient-exhausted
50    /// failures and orphan re-routing failures.
51    pub failed: HashMap<String, String>,
52}
53
54/// Configuration for one grouped cascade copy.
55pub struct CascadeCopyConfig<'a> {
56    /// Where the closures originate. The host running cast IS the seed
57    /// — `seed_addr` is just for display; the [`NixCopyExecutor`] uses
58    /// local `nix copy` for any edge whose source is the seed.
59    pub seed_addr: String,
60    /// All targets from this deploy run. Will be grouped by toplevel.
61    pub targets: Vec<CascadeCopyTarget>,
62    /// Children-per-node in the F-ary tree. 2 = binary; bigger trades
63    /// per-source bandwidth contention for fewer rounds. Default 2.
64    pub fanout: u32,
65    /// Per-edge `nix copy` timeout. Default 5min.
66    pub timeout: Duration,
67    /// Optional event sink for live UI. Use [`NullSink`] for headless.
68    /// Reused across all groups (the renderer can handle multiple
69    /// cascade runs back-to-back, though it'll show them sequentially).
70    pub events: Option<&'a dyn EventSink>,
71}
72
73impl<'a> CascadeCopyConfig<'a> {
74    pub fn new(seed_addr: impl Into<String>, targets: Vec<CascadeCopyTarget>) -> Self {
75        Self {
76            seed_addr: seed_addr.into(),
77            targets,
78            fanout: 2,
79            timeout: Duration::from_secs(300),
80            events: None,
81        }
82    }
83
84    pub fn fanout(mut self, n: u32) -> Self {
85        self.fanout = n.max(1);
86        self
87    }
88
89    pub fn timeout(mut self, t: Duration) -> Self {
90        self.timeout = t;
91        self
92    }
93
94    pub fn events(mut self, sink: &'a dyn EventSink) -> Self {
95        self.events = Some(sink);
96        self
97    }
98}

100/// Group `targets` by their `toplevel_path`, then run one cascade per
101/// group. Returns the union of results across all groups.
102///
103/// # Behavior
104///
105/// - Empty `targets` → empty result, no work.
106/// - All targets share one toplevel → 1 cascade, N targets.
107/// - All targets unique toplevels → N cascades, each of 1 target
108///   (degenerates to direct copy — same wall-time as today's serial
109///   loop, no regression).
110///
111/// # Strategy
112///
113/// Currently hardcoded to [`LevelTreeFanOut`]. The `MaxBottleneckSpanning`
114/// / `SteinerGreedy` strategies require a populated `NetworkProfile`
115/// which we don't gather for production deploys (would need an active
116/// bandwidth probe step). When that lands, swap the strategy here.
117pub fn cascade_copy_grouped(cfg: CascadeCopyConfig<'_>) -> CascadeCopyResult {
118    let mut result = CascadeCopyResult::default();
119    if cfg.targets.is_empty() {
120        return result;
121    }
122
123    // Group targets by their toplevel path.
124    let mut groups: HashMap<String, Vec<CascadeCopyTarget>> = HashMap::new();
125    for t in cfg.targets {
126        groups.entry(t.toplevel_path.clone()).or_default().push(t);
127    }
128
129    let strategy = LevelTreeFanOut::new(cfg.fanout);
130    let null_sink = NullSink;
131    let events: &dyn EventSink = cfg.events.unwrap_or(&null_sink);
132
133    for (toplevel, group) in groups {
134        run_one_group(
135            &cfg.seed_addr,
136            &toplevel,
137            group,
138            &strategy,
139            events,
140            cfg.timeout,
141            &mut result,
142        );
143    }
144
145    result
146}
147
148fn run_one_group(
149    seed_addr: &str,
150    toplevel: &str,
151    group: Vec<CascadeCopyTarget>,
152    strategy: &LevelTreeFanOut,
153    events: &dyn EventSink,
154    timeout: Duration,
155    result: &mut CascadeCopyResult,
156) {
157    // Build NodeIds: seed at NodeId(0), targets at NodeId(1)..
158    let mut alloc = NodeIdAlloc::new();
159    let seed_id = alloc.alloc();
160
161    let mut cascade_nodes: Vec<CascadeNode> =
162        vec![CascadeNode::new(seed_id, seed_addr.to_string())];
163    let mut addrs: HashMap<NodeId, String> = HashMap::new();
164    addrs.insert(seed_id, seed_addr.to_string());
165
166    let mut id_to_host: HashMap<NodeId, String> = HashMap::new();
167
168    for t in &group {
169        let id = alloc.alloc();
170        cascade_nodes.push(CascadeNode::new(id, t.ssh_addr.clone()));
171        addrs.insert(id, t.ssh_addr.clone());
172        id_to_host.insert(id, t.host_name.clone());
173    }
174
175    let mut seeded = HashSet::new();
176    seeded.insert(seed_id);
177
178    let executor = NixCopyExecutor::new(addrs, toplevel.to_string(), seed_id).with_timeout(timeout);
179
180    let cascade_result = Cascade::new()
181        .nodes(cascade_nodes)
182        .seeded(seeded)
183        .network(NetworkProfile::default())
184        .strategy(strategy)
185        .executor(&executor)
186        .events(events)
187        .run();
188
189    // Map NodeIds back to host names.
190    for id in &cascade_result.converged {
191        if let Some(host) = id_to_host.get(id) {
192            // The seed isn't a deploy target; skip it.
193            if *id != seed_id {
194                result.copied.push(host.clone());
195            }
196        }
197    }
198    if let Some(err) = cascade_result.failed {
199        let msg = format!("{}", err);
200        // Dedupe affected nodes — a single subtree-aggregate can name
201        // the same host multiple times if it bubbled up across rounds.
202        let mut seen = HashSet::new();
203        for affected_id in err.affected_nodes() {
204            if !seen.insert(affected_id) {
205                continue;
206            }
207            if let Some(host) = id_to_host.get(&affected_id) {
208                result.failed.insert(host.clone(), msg.clone());
209            }
210        }
211    }
212}

218    fn t(name: &str, addr: &str, tl: &str) -> CascadeCopyTarget {
219        CascadeCopyTarget {
220            host_name: name.into(),
221            ssh_addr: addr.into(),

226    #[test]
227    fn empty_targets_returns_empty() {
228        let cfg = CascadeCopyConfig::new("root@seed", vec![]);
229        let r = cascade_copy_grouped(cfg);
230        assert!(r.copied.is_empty());
231        assert!(r.failed.is_empty());
232    }

234    // End-to-end behavior is exercised by the cascade_executor tests
235    // (which spawn real subprocesses). Group bookkeeping is exercised
236    // by the empty-targets case + manual smoke via cascade-copy bin.

238    // leaving for when somebody hits the "what if nix copy returned
239    // <weird thing>" question and needs the seam.
