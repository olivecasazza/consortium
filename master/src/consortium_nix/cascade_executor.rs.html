1//! Production [`RoundExecutor`] implementations — wrap real `nix copy`
2//! subprocesses, runnable against actual nixlab hosts.
3//!
4//! [`NixCopyExecutor`] is the realistic counterpart to
5//! [`consortium_fanout_sim::DeterministicExecutor`]. The sim does
6//! `closure_size / bandwidth + latency` math; this one shells out to
7//! `nix copy --no-check-sigs --to ssh-ng://user@host store_path` for
8//! every (src, tgt) edge in a round, in parallel via `std::thread`.
9//!
10//! ## Edge-source handling
11//!
12//! For an edge `(src, tgt)` with the cascade's seed at `seed_id`:
13//! - If `src == seed_id`: run `nix copy ...` LOCALLY (the host running
14//!   cascade-copy IS the seed)
15//! - Otherwise: SSH into `src` and run `nix copy ...` THERE (the source
16//!   forwards the closure it received in a prior round)
17//!
18//! ## Trust + signing
19//!
20//! `--no-check-sigs` is passed because closures built locally are NOT
21//! signed by a key the remote trusts. Trust boundary is the SSH
22//! connection itself (root-over-ssh deploy assumption — same as
23//! `consortium_nix::copy::copy_closure`).
24//!
25//! ## Failure mapping
26//!
27//! `nix copy` exit status maps to:
28//! - exit 0 → `Ok(elapsed_duration)`
29//! - non-zero with stderr containing "Connection refused"/"Permission
30//!   denied" → `CascadeError::SshHandshake` (permanent — no retry from
31//!   alt source will help if the tgt itself is unreachable)
32//! - non-zero otherwise → `CascadeError::Copy` (transient — retry from
33//!   alt source might succeed, e.g. if it was source-side bandwidth)
34

43/// Real-world `RoundExecutor` that drives `nix copy` over SSH.
44pub struct NixCopyExecutor {
45    /// NodeId → SSH address (e.g. `"root@hp01"` or `"olive@seir"`).
46    /// Seed node also has an entry here for symmetry, even though
47    /// edges originating from it run locally.
48    pub addrs: HashMap<NodeId, String>,
49    /// The store path being distributed, e.g. `/nix/store/xxx-foo-1.0`.
50    pub store_path: String,
51    /// NodeId of the seed — edges originating from it run via local
52    /// `nix copy`; all other src edges run via `ssh <src> 'nix copy …'`.
53    pub seed: NodeId,
54    /// Per-edge subprocess timeout. Cascade halts the edge if the SSH
55    /// or `nix copy` hasn't returned by this point — typically the
56    /// remote is unreachable. Default 5 minutes.
57    pub timeout: Duration,
58}
59
60impl NixCopyExecutor {
61    pub fn new(
62        addrs: HashMap<NodeId, String>,
63        store_path: impl Into<String>,
64        seed: NodeId,
65    ) -> Self {
66        Self {
67            addrs,
68            store_path: store_path.into(),
69            seed,
70            timeout: Duration::from_secs(300),
71        }
72    }
73
74    pub fn with_timeout(mut self, t: Duration) -> Self {
75        self.timeout = t;
76        self
77    }
78
79    /// Run a single edge: src copies the closure to tgt. Blocks until
80    /// the subprocess completes, the subprocess errors out, or the
81    /// per-edge timeout fires.
82    ///
83    /// Returns `Ok(elapsed)` on success, `Err(CascadeError)` otherwise.
84    fn run_edge(&self, src: NodeId, tgt: NodeId) -> Result<Duration, CascadeError> {
85        let Some(tgt_addr) = self.addrs.get(&tgt) else {
86            return Err(CascadeError::Copy {
87                node: tgt,
88                stderr: format!("no SSH address registered for tgt {tgt}"),
89            });
90        };
91        let store_uri = format!("ssh-ng://{tgt_addr}");
92
93        let started = Instant::now();
94        let cmd_result = if src == self.seed {
95            // Local nix copy from the seed.
96            Command::new("nix")
97                .args([
98                    "copy",
99                    "--no-check-sigs",
100                    "--to",
101                    &store_uri,
102                    &self.store_path,
103                ])
104                .output()
105        } else {
106            // SSH into src and have it run nix copy.
107            let Some(src_addr) = self.addrs.get(&src) else {
108                return Err(CascadeError::Copy {
109                    node: tgt,
110                    stderr: format!("no SSH address registered for src {src}"),
111                });
112            };
113            // Build the remote command. Quote the store path so spaces
114            // (rare but possible) don't break parsing.
115            let remote_cmd = format!(
116                "nix copy --no-check-sigs --to {} {}",
117                shell_escape(&store_uri),
118                shell_escape(&self.store_path),
119            );
120            Command::new("ssh")
121                .args([
122                    "-o",
123                    "BatchMode=yes",
124                    "-o",
125                    "StrictHostKeyChecking=accept-new",
126                ])
127                .arg(src_addr)
128                .arg(remote_cmd)
129                .output()
130        };
131
132        let elapsed = started.elapsed();
133        match cmd_result {
134            Ok(output) if output.status.success() => Ok(elapsed),
135            Ok(output) => {
136                let stderr = String::from_utf8_lossy(&output.stderr).to_string();
137                Err(classify_copy_error(tgt, src, &stderr))
138            }
139            Err(io_err) => Err(CascadeError::Copy {
140                node: tgt,
141                stderr: format!("subprocess spawn failed: {io_err}"),
142            }),
143        }
144    }
145}

147impl RoundExecutor for NixCopyExecutor {
148    fn dispatch(
149        &self,
150        _nodes: &[CascadeNode],
151        edges: &[(NodeId, NodeId)],
152        _net: &NetworkProfile,
153    ) -> HashMap<(NodeId, NodeId), Result<Duration, CascadeError>> {
154        // Spawn one thread per edge; collect via channel so we don't
155        // need to box the closures or hold thread handles.
156        let (tx, rx) = mpsc::channel();
157        let n = edges.len();
158        thread::scope(|scope| {
159            for &(src, tgt) in edges {
160                let tx = tx.clone();
161                let me = &*self;
162                scope.spawn(move || {
163                    let outcome = me.run_edge(src, tgt);
164                    let _ = tx.send(((src, tgt), outcome));
165                });
166            }
167        });
168        drop(tx); // close the channel so the rx loop terminates
169        let mut out = HashMap::with_capacity(n);
170        while let Ok(item) = rx.try_recv() {
171            out.insert(item.0, item.1);
172        }
173        out
174    }
175}

177/// Map a non-zero `nix copy` stderr to the right `CascadeError` variant.
178/// Permanent vs transient distinction matters for orphan re-routing
179/// (see cascade.rs's `is_transient()` discussion).
180fn classify_copy_error(tgt: NodeId, src: NodeId, stderr: &str) -> CascadeError {
181    let lower = stderr.to_lowercase();
182    if lower.contains("connection refused")
183        || lower.contains("connection timed out")
184        || lower.contains("no route to host")
185        || lower.contains("permission denied")
186        || lower.contains("host key verification failed")
187    {
188        // Target host is permanently unreachable — orphan re-routing
189        // should kick in for any descendants in level-tree.
190        CascadeError::SshHandshake {
191            node: tgt,
192            parent: src,
193        }
194    } else {
195        // Default to transient — could be source-side bandwidth, a
196        // flaky relay, or a substituter being slow. Retry from an
197        // alternate source on the next round may succeed.
198        CascadeError::Copy {
199            node: tgt,
200            stderr: stderr.lines().take(5).collect::<Vec<_>>().join("\n"),
201        }
202    }
203}

205/// Minimal POSIX shell escape — wraps in single quotes, doubles any
206/// embedded single quote. Safe for `ssh remote 'cmd ...'` invocations.
207fn shell_escape(s: &str) -> String {
208    let mut out = String::with_capacity(s.len() + 2);
209    out.push('\'');
210    for c in s.chars() {
211        if c == '\'' {
212            out.push_str(r#"'\''"#);
213        } else {
214            out.push(c);
215        }
216    }
217    out.push('\'');
218    out
219}

221#[cfg(test)]
222mod tests {

224
225    #[test]
226    fn shell_escape_basic() {
227        assert_eq!(shell_escape("foo"), "'foo'");
228        assert_eq!(shell_escape("/nix/store/xxx-foo"), "'/nix/store/xxx-foo'");
229        assert_eq!(shell_escape("with 'quote'"), r#"'with '\''quote'\'''"#);
230        assert_eq!(shell_escape(""), "''");
231    }
232
233    #[test]
234    fn classify_known_permanent_errors() {
235        let perm_cases = [
236            "ssh: connect to host hp01 port 22: Connection refused",
237            "ssh: connect to host hp01 port 22: Connection timed out",
238            "Permission denied (publickey).",
239            "Host key verification failed.",
240            "ssh: connect to host hp01 port 22: No route to host",
241        ];
242        for stderr in perm_cases {
243            let err = classify_copy_error(NodeId(1), NodeId(0), stderr);
244            assert!(
245                matches!(err, CascadeError::SshHandshake { .. }),
246                "expected SshHandshake for stderr={stderr:?}, got {err:?}"
247            );
248        }
249    }

251    #[test]
252    fn classify_transient_errors_default_to_copy() {
253        let cases = [
254            "error: writing to file: No space left on device",
255            "warning: substituter 'https://cache.nixos.org' returned HTTP 503",
256            "some random stderr nobody categorized",
257        ];
258        for stderr in cases {
259            let err = classify_copy_error(NodeId(1), NodeId(0), stderr);
260            assert!(
261                matches!(err, CascadeError::Copy { .. }),
262                "expected Copy (transient) for stderr={stderr:?}, got {err:?}"
263            );
264        }
