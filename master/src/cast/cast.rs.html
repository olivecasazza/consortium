1//! cast — NixOS deployment orchestration CLI.
2//!
3//! Usage:
4//!   cast deploy --on hp[01-03] switch     # build, copy, activate
5//!   cast build --on contra                # build only
6//!   cast eval --on mm[01-05]              # show what would be built
7//!   cast health                           # probe all builders
8//!   cast status --on hp[01-03]            # show current system versions
9

19/// cast — NixOS deployment orchestration powered by consortium.
20#[derive(Parser)]
21#[command(name = "cast", version, about)]
22struct Args {
23    /// Path to fleet configuration JSON file.
24    #[arg(short, long, default_value = "fleet.json")]
25    config: PathBuf,
26
27    /// Override the flake URI from fleet config (e.g. /home/user/nixlab or github:user/repo).
28    #[arg(long)]
29    flake: Option<String>,
30
31    #[command(subcommand)]
32    command: Commands,
33}
34
35#[derive(Subcommand)]
36enum Commands {
37    /// Evaluate which hosts need deployment.
38    Eval {
39        /// Target nodes (bracket notation, e.g., hp[01-03]).
40        #[arg(short = 'w', long = "on")]
41        on: Option<String>,
42
43        /// Target nodes by tag.
44        #[arg(short = 'g', long = "tag")]
45        tag: Vec<String>,
46    },
47
48    /// Build system closures.
49    Build {
50        /// Target nodes (bracket notation).
51        #[arg(short = 'w', long = "on")]
52        on: Option<String>,
53
54        /// Target nodes by tag.
55        #[arg(short = 'g', long = "tag")]
56        tag: Vec<String>,
57
58        /// Use distributed builders.
59        #[arg(long)]
60        builders: bool,
61
62        /// Maximum parallel builds.
63        #[arg(short = 'f', long = "fanout", default_value = "4")]
64        fanout: usize,
65    },
66
67    /// Deploy to targets (build + copy + activate).
68    Deploy {
69        /// Target nodes (bracket notation).
70        #[arg(short = 'w', long = "on")]
71        on: Option<String>,
72
73        /// Target nodes by tag.
74        #[arg(short = 'g', long = "tag")]
75        tag: Vec<String>,
76
77        /// Deployment action.
78        #[arg(default_value = "switch")]
79        action: String,
80
81        /// Use distributed builders.
82        #[arg(long)]
83        builders: bool,
84
85        /// Maximum parallel operations.
86        #[arg(short = 'f', long = "fanout", default_value = "4")]
87        fanout: usize,
88    },
89
90    /// Probe builder health.
91    Health,
92
93    /// Show current system versions on targets.
94    Status {
95        /// Target nodes (bracket notation).
96        #[arg(short = 'w', long = "on")]
97        on: Option<String>,
98
99        /// Target nodes by tag.
100        #[arg(short = 'g', long = "tag")]
101        tag: Vec<String>,
102    },
103}
104
105fn main() {
106    let args = Args::parse();
107
108    let mut config = match FleetConfig::from_file(&args.config) {
109        Ok(c) => c,
110        Err(e) => {
111            eprintln!("error: {}", e);
112            eprintln!(
113                "hint: build fleet config first with: nix build .#fleet-config && cp result fleet.json"
114            );
115            process::exit(1);
116        }
117    };
118
119    // Override flake URI if --flake was provided
120    if let Some(ref flake) = args.flake {
121        config.flake_uri = flake.clone();
122    }
123
124    let result = match args.command {
125        Commands::Eval { on, tag } => cmd_eval(&config, on.as_deref(), &tag),
126        Commands::Build {
127            on,
128            tag,
129            builders,
130            fanout,
131        } => cmd_deploy(&config, on.as_deref(), &tag, "build", builders, fanout),
132        Commands::Deploy {
133            on,
134            tag,
135            action,
136            builders,
137            fanout,
138        } => cmd_deploy(&config, on.as_deref(), &tag, &action, builders, fanout),
139        Commands::Health => cmd_health(&config),
140        Commands::Status { on, tag } => cmd_status(&config, on.as_deref(), &tag),
141    };
142
143    if let Err(e) = result {
144        eprintln!("error: {}", e);
145        process::exit(1);
146    }
147}

149/// Resolve target nodes from --on and --tag flags, falling back to all nodes.
150fn resolve_targets(
151    config: &FleetConfig,
152    on: Option<&str>,
153    tags: &[String],
154) -> anyhow::Result<Vec<String>> {
155    let mut targets = Vec::new();
156
157    if let Some(pattern) = on {
158        let ns = NodeSet::parse(pattern)
159            .map_err(|e| anyhow::anyhow!("invalid node pattern '{}': {}", pattern, e))?;
160        targets.extend(ns.iter());
161    }
162
163    if !tags.is_empty() {
164        let tag_nodes = config.nodes_by_tags(tags);
165        for node in tag_nodes {
166            if !targets.contains(&node.name) {
167                targets.push(node.name.clone());
168            }
169        }
170    }
171
172    // Default to all nodes if nothing specified
173    if targets.is_empty() {
174        targets = config.node_names();
175    }
176
177    // Validate all targets exist in config
178    for name in &targets {
179        if !config.nodes.contains_key(name) {
180            anyhow::bail!(
181                "unknown node '{}' (available: {})",
182                name,
183                config.node_names().join(", ")
184            );
185        }
186    }
187
188    Ok(targets)
189}

191fn cmd_eval(config: &FleetConfig, on: Option<&str>, tags: &[String]) -> anyhow::Result<()> {
192    let targets = resolve_targets(config, on, tags)?;
193
194    println!("Evaluating {} host(s):", targets.len());
195    for name in &targets {
196        let node = &config.nodes[name];
197        println!(
198            "  {} → {} ({}@{}, tags: [{}])",
199            name,
200            node.system,
201            node.target_user,
202            node.target_host,
203            node.tags.join(", ")
204        );
205        if let Some(ref drv) = node.drv_path {
206            println!("    drv: {}", drv);
207        }
208    }
209
210    Ok(())
211}

213fn cmd_deploy(
214    config: &FleetConfig,
215    on: Option<&str>,
216    tags: &[String],
217    action_str: &str,
218    use_builders: bool,
219    fanout: usize,
220) -> anyhow::Result<()> {
221    let targets = resolve_targets(config, on, tags)?;
222    let action: DeployAction = action_str.parse().map_err(|_| {
223        anyhow::anyhow!(
224            "invalid action '{}' (try: switch, boot, test, dry-activate, build)",
225            action_str
226        )
227    })?;
228
229    println!(
230        "Deploying {} host(s) with action '{}':",
231        targets.len(),
232        action
233    );
234    for name in &targets {
235        println!("  {}", name);
236    }
237
238    let report = consortium_nix::deploy(config, &targets, action, fanout, use_builders)?;
239
240    println!();
241    if report.is_success() {
242        println!("Deployment successful!");
243        println!("  Built: {}", report.built.len());
244        if action != DeployAction::Build {
245            println!("  Copied: {}", report.copied.len());
246            println!("  Activated: {}", report.activated.len());
247        }
248    } else {
249        println!(
250            "Deployment completed with {} failure(s):",
251            report.failure_count()
252        );
253        for (name, err) in &report.build_failures {
254            eprintln!("  build failed on {}: {}", name, err);
255        }
256        for (name, err) in &report.copy_failures {
257            eprintln!("  copy failed to {}: {}", name, err);
258        }
259        for (name, err) in &report.activation_failures {
260            eprintln!("  activation failed on {}: {}", name, err);
261        }
262        process::exit(1);
263    }
264
265    Ok(())
266}

268fn cmd_health(config: &FleetConfig) -> anyhow::Result<()> {
269    if config.builders.is_empty() {
270        println!("No builders configured.");
271        return Ok(());
272    }
273
274    println!("Probing {} builder(s):", config.builders.len());
275
276    let statuses = health::check_builders(config);
277    let healthy_count = statuses.iter().filter(|s| s.healthy).count();
278
279    for status in &statuses {
280        let icon = if status.healthy { "ok" } else { "FAIL" };
281        let latency = status
282            .latency_ms
283            .map(|ms| format!(" ({}ms)", ms))
284            .unwrap_or_default();
285        let error = status
286            .error
287            .as_ref()
288            .map(|e| format!(" - {}", e))
289            .unwrap_or_default();
290
291        println!(
292            "  [{}] {}@{}{}{} (jobs:{}, speed:{}x)",
293            icon,
294            status.builder.user,
295            status.builder.host,
296            latency,
297            error,
298            status.builder.max_jobs,
299            status.builder.speed_factor,
300        );
301    }
302
303    println!("\n{}/{} builders healthy", healthy_count, statuses.len());
304
305    Ok(())
306}

308fn cmd_status(config: &FleetConfig, on: Option<&str>, tags: &[String]) -> anyhow::Result<()> {
309    let targets = resolve_targets(config, on, tags)?;
310
311    println!("Querying {} host(s):", targets.len());
312
313    for name in &targets {
314        let node = &config.nodes[name];
315        match consortium_nix::eval::query_current_system(&node.target_host, &node.target_user) {
316            Ok(Some(path)) => println!("  {} → {}", name, path),
317            Ok(None) => println!("  {} → (unknown)", name),
318            Err(e) => println!("  {} → error: {}", name, e),
319        }
320    }
321
322    Ok(())
323}
