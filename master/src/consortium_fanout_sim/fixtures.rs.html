1//! Reproducible generators for cascade scenarios.
2//!
3//! All generators take an `&mut StdRng` (or `ChaCha8Rng`) seeded from
4//! the scenario's master seed, so a given `(seed, descriptor)` pair
5//! produces deterministic output.
6//!
7//! ## Knobs
8//!
9//! - [`SeedDistribution`] — which subset of nodes starts with the closure
10//! - [`BandwidthDistribution`] — how per-edge bandwidth varies
11//! - [`FailureSchedule`] — when and where to inject failures during the run
12//!
13//! Each is a sum-type so callers can mix and match without baking a
14//! specific distribution into the simulator.
15

25// ============================================================================
26// Seed distribution — which nodes start with the closure
27// ============================================================================
28
29/// How to choose the initial set of nodes that have the closure at
30/// cascade start.
31#[derive(Debug, Clone)]
32pub enum SeedDistribution {
33    /// Exactly one node (the build host) starts with the closure.
34    /// Always picks `NodeId(0)` for reproducibility.
35    Single,
36    /// `fraction` of all nodes are pre-seeded, chosen uniformly at random.
37    /// `fraction` clamped to `[0, 1]`.
38    Random { fraction: f64 },
39    /// Exactly `count` nodes pre-seeded (chosen uniformly at random).
40    Count(usize),
41    /// Caller-specified set of nodes (escape hatch for hand-crafted scenarios).
42    Explicit(HashSet<NodeId>),
43}
44
45impl SeedDistribution {
46    pub fn sample(&self, rng: &mut ChaCha8Rng, n_nodes: u32) -> HashSet<NodeId> {
47        match self {
48            SeedDistribution::Single => {
49                let mut s = HashSet::new();
50                if n_nodes > 0 {
51                    s.insert(NodeId(0));
52                }
53                s
54            }
55            SeedDistribution::Random { fraction } => {
56                let f = fraction.clamp(0.0, 1.0);
57                let count = ((n_nodes as f64) * f).round() as usize;
58                Self::Count(count.max(1).min(n_nodes as usize)).sample(rng, n_nodes)
59            }
60            SeedDistribution::Count(c) => {
61                let count = (*c).min(n_nodes as usize);
62                let mut ids: Vec<NodeId> = (0..n_nodes).map(NodeId).collect();
63                ids.shuffle(rng);
64                ids.into_iter().take(count).collect()
65            }
66            SeedDistribution::Explicit(s) => s.clone(),
67        }
68    }
69}

71// ============================================================================
72// Bandwidth distribution
73// ============================================================================
74
75/// How per-edge bandwidth varies. Generators populate the
76/// [`NetworkProfile::bandwidth`] map; edges absent from the map fall
77/// through to the executor's default.
78#[derive(Debug, Clone)]
79pub enum BandwidthDistribution {
80    /// Every edge gets the same bandwidth (bytes/sec).
81    Uniform(u64),
82    /// Each edge picks from `{slow, fast}` with `fast_fraction` of
83    /// edges getting `fast` bandwidth, the rest `slow`.
84    Bimodal {
85        slow: u64,
86        fast: u64,
87        fast_fraction: f64,
88    },
89    /// Each edge sampled from a discrete distribution: `(weight, bytes/sec)`.
90    /// Weights are relative.
91    Discrete(Vec<(u32, u64)>),
92}
93
94impl BandwidthDistribution {
95    pub fn populate(&self, rng: &mut ChaCha8Rng, net: &mut NetworkProfile, n_nodes: u32) {
96        match self {
97            BandwidthDistribution::Uniform(bw) => {
98                for src in 0..n_nodes {
99                    for tgt in 0..n_nodes {
100                        if src == tgt {
101                            continue;
102                        }
103                        net.bandwidth.insert((NodeId(src), NodeId(tgt)), *bw);
104                    }
105                }
106            }
107            BandwidthDistribution::Bimodal {
108                slow,
109                fast,
110                fast_fraction,
111            } => {
112                let f = fast_fraction.clamp(0.0, 1.0);
113                for src in 0..n_nodes {
114                    for tgt in 0..n_nodes {
115                        if src == tgt {
116                            continue;
117                        }
118                        let bw = if rng.gen::<f64>() < f { *fast } else { *slow };
119                        net.bandwidth.insert((NodeId(src), NodeId(tgt)), bw);
120                    }
121                }
122            }
123            BandwidthDistribution::Discrete(weights) => {
124                let dist = WeightedIndex::new(weights.iter().map(|(w, _)| *w))
125                    .expect("BandwidthDistribution::Discrete needs at least one weight > 0");
126                for src in 0..n_nodes {
127                    for tgt in 0..n_nodes {
128                        if src == tgt {
129                            continue;
130                        }
131                        let idx = dist.sample(rng);
132                        let bw = weights[idx].1;
133                        net.bandwidth.insert((NodeId(src), NodeId(tgt)), bw);
134                    }
135                }
136            }
137        }
138    }
139}

141// ============================================================================
142// Uplink distribution — per-node link capacities
143// ============================================================================
144
145/// How to assign per-node link capacities. Generators populate the
146/// [`NetworkProfile::nodes`] map, which engages contention modeling in
147/// [`NetworkProfile::effective_bandwidth`].
148///
149/// Nodes absent from the map are treated as having infinite uplink/downlink
150/// (degenerate to per-edge bandwidth only — same as the simple model).
151#[derive(Debug, Clone)]
152pub enum UplinkDistribution {
153    /// Every node gets the same uplink. Downlink = 4 × uplink, matching
154    /// the asymmetric-link assumption in `NetworkBuilder::uplinks_uniform`.
155    Uniform(u64),
156    /// Bimodal: data-center nodes (fast) and residential nodes (slow).
157    /// `fast_fraction` of nodes get the fast uplink; the rest get slow.
158    /// Downlink = 4 × uplink for each tier.
159    Bimodal {
160        slow: u64,
161        fast: u64,
162        fast_fraction: f64,
163    },
164    /// Caller-specified per-node specs (escape hatch for hand-crafted scenarios).
165    Explicit(HashMap<NodeId, NodeSpec>),
166}
167
168impl UplinkDistribution {
169    /// Populate `net.nodes` with per-node specs for `0..n_nodes`.
170    pub fn populate(&self, rng: &mut ChaCha8Rng, net: &mut NetworkProfile, n_nodes: u32) {
171        match self {
172            UplinkDistribution::Uniform(uplink) => {
173                for i in 0..n_nodes {
174                    net.nodes.insert(
175                        NodeId(i),
176                        NodeSpec {
177                            uplink: *uplink,
178                            downlink: uplink.saturating_mul(4),
179                        },
180                    );
181                }
182            }
183            UplinkDistribution::Bimodal {
184                slow,
185                fast,
186                fast_fraction,
187            } => {
188                let f = fast_fraction.clamp(0.0, 1.0);
189                for i in 0..n_nodes {
190                    let uplink = if rng.gen::<f64>() < f { *fast } else { *slow };
191                    net.nodes.insert(
192                        NodeId(i),
193                        NodeSpec {
194                            uplink,
195                            downlink: uplink.saturating_mul(4),
196                        },
197                    );
198                }
199            }
200            UplinkDistribution::Explicit(specs) => {
201                net.nodes.extend(specs.iter().map(|(k, v)| (*k, *v)));
202            }
203        }
204    }
205}

207// ============================================================================
208// Failure schedule
209// ============================================================================
210
211/// Injects deterministic failures into the cascade. The
212/// `DeterministicExecutor` calls `failure_for(round, src, tgt)` for
213/// every edge it dispatches; non-`None` returns become the edge's
214/// outcome.
215#[derive(Debug, Clone, Default)]
216pub enum FailureSchedule {
217    /// No failures.
218    #[default]
219    None,
220    /// Fail every edge whose target is `node`, starting at `round`.
221    KillNodeAtRound { node: NodeId, round: u32 },
222    /// Fail the specific `(src, tgt)` edge at `round`.
223    PartitionAtRound {
224        src: NodeId,
225        tgt: NodeId,
226        round: u32,
227    },
228    /// Fail each edge with probability `fraction` (0.0 to 1.0). The
229    /// outcome is deterministic — same `seed`, `round`, `src`, `tgt`
230    /// produce the same fail/succeed decision via a hash of all four,
231    /// so reproducing a "weird" failure pattern is just a matter of
232    /// re-running with the same seed.
233    Random { fraction: f64, seed: u64 },
234    /// Fail edges per a hand-built table: `(round, src, tgt) -> Error`.
235    Explicit(HashMap<(u32, NodeId, NodeId), CascadeError>),
236    /// Multiple failure schedules composed (any one matching = fail).
237    All(Vec<FailureSchedule>),
238}
239
240impl FailureSchedule {
241    pub fn failure_for(&self, round: u32, src: NodeId, tgt: NodeId) -> Option<CascadeError> {
242        match self {
243            FailureSchedule::None => None,
244            FailureSchedule::KillNodeAtRound { node, round: r } => {
245                if round >= *r && tgt == *node {
246                    // Permanent failure: the node is dead. Use
247                    // Activation error (signals "target node failed
248                    // to activate the closure") so the coordinator
249                    // marks it in failed_nodes. Orphan re-routing
250                    // then kicks in for any descendants in level-tree.
251                    Some(CascadeError::Activation {
252                        node: tgt,
253                        stage: "killed",
254                    })
255                } else {
256                    None
257                }
258            }
259            FailureSchedule::PartitionAtRound {
260                src: s,
261                tgt: t,
262                round: r,
263            } => {
264                if round >= *r && src == *s && tgt == *t {
265                    Some(CascadeError::Partitioned { src, tgt })
266                } else {
267                    None
268                }
269            }
270            FailureSchedule::Random { fraction, seed } => {
271                // Deterministic per-edge decision: mix (seed, round, src,
272                // tgt) into a u64 with FxHash-style XOR/multiply, then
273                // map to [0, 1) and compare against fraction. Same
274                // inputs → same outcome, so a failure pattern is
275                // reproducible from the seed. Different rounds get
276                // different hashes, so a retry has a fresh chance.
277                let mut state: u64 = *seed;
278                state ^= (round as u64).wrapping_mul(0x9E37_79B9_7F4A_7C15);
279                state ^= (src.0 as u64).wrapping_mul(0xBF58_476D_1CE4_E5B9);
280                state ^= (tgt.0 as u64).wrapping_mul(0x94D0_49BB_1331_11EB);
281                state = state.wrapping_mul(0x2545_F491_4F6C_DD1D);
282                state ^= state >> 33;
283                // Map to [0.0, 1.0): take high 53 bits (after the >>33
284                // mixing has scrambled them), divide by 2^53.
285                let p = ((state >> 11) as f64) * (1.0 / ((1u64 << 53) as f64));
286                if p < fraction.clamp(0.0, 1.0) {
287                    Some(CascadeError::Copy {
288                        node: tgt,
289                        stderr: format!(
290                            "random failure (seed={seed} round={round} src={src} tgt={tgt})"
291                        ),
292                    })
293                } else {
294                    None
295                }
296            }
297            FailureSchedule::Explicit(map) => map.get(&(round, src, tgt)).cloned(),
298            FailureSchedule::All(schedules) => schedules
299                .iter()
300                .find_map(|s| s.failure_for(round, src, tgt)),
301        }
302    }
303}

305// ============================================================================
306// Latency distribution (helper, not first-class)
307// ============================================================================
308
309/// Sprinkle uniform latency across all edges. Useful when bandwidth
310/// alone isn't enough to differentiate strategies (latency dominates
311/// for tiny closures).
312pub fn populate_uniform_latency(net: &mut NetworkProfile, latency: Duration, n_nodes: u32) {
313    for src in 0..n_nodes {
314        for tgt in 0..n_nodes {
315            if src == tgt {
316                continue;
317            }
318            net.latency.insert((NodeId(src), NodeId(tgt)), latency);
319        }
320    }
321}

375    #[test]
376    fn random_failure_schedule_respects_fraction_at_scale() {
377        // Sample 10000 edges at fraction=0.3; expect ~3000 failures.
378        let s = FailureSchedule::Random {
379            fraction: 0.3,
380            seed: 0xc0ffee,
381        };
382        let mut fails = 0;
383        let n_samples = 10_000u32;
384        for i in 0..n_samples {
385            // Use varied (round, src, tgt) so the hash mixes properly.
386            let round = i / 100;
387            let src = NodeId(i % 50);
388            let tgt = NodeId(i % 47);
389            if s.failure_for(round, src, tgt).is_some() {
390                fails += 1;
391            }
392        }
393        let observed = fails as f64 / n_samples as f64;
394        assert!(
395            (observed - 0.3).abs() < 0.03,
396            "expected ~30% failure rate; got {observed:.2}% over {n_samples} samples"
397        );
398    }
