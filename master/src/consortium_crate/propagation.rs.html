2//!
3//! Rust implementation of `ClusterShell.Propagation`.
4//!
5//! This module provides:
6//! - [`PropagationTreeRouter`]: Routes commands through the topology tree by
7//!   computing next-hop gateways and managing fanout/load-balancing.
8//! - [`PropagationChannel`]: State machine for gateway-to-gateway channel

23/// Errors raised during route resolving operations.
24#[derive(Error, Debug)]
25pub enum RouteResolvingError {
26    /// Invalid root or gateway node.
27    #[error("Invalid root or gateway node: {0}")]
28    InvalidRoot(String),
29
30    /// No route available to the destination.
31    #[error("No route available to {0}")]
32    NoRoute(String),
33
34    /// Destination host is unreachable.
35    #[error("Invalid destination: {0}, host is unreachable")]
36    Unreachable(String),
37
38    /// Source equals destination.
39    #[error("Invalid resolution request: {0} -> {1}")]
40    SameSourceDest(String, String),
41
42    /// Topology error.
43    #[error("Topology error: {0}")]
44    Topology(#[from] TopologyError),
45
46    /// NodeSet error.
47    #[error("NodeSet error: {0}")]
48    NodeSet(#[from] crate::node_set::NodeSetError),
49}

55/// A routing table entry: (reachable network, next-hop gateways).
56#[derive(Debug, Clone)]
57pub struct RouteEntry {
58    /// Set of all nodes reachable through this route (transitive children).
59    network: NodeSet,
60    /// Direct next-hop gateway(s) for this route.
61    nexthops: NodeSet,
62}
63
64/// Performs route resolving operations within a propagation tree.
65///
66/// The router parses the topology tree to generate a routing table, then
67/// provides [`next_hop()`](PropagationTreeRouter::next_hop) and
68/// [`dispatch()`](PropagationTreeRouter::dispatch) methods for directing
69/// messages to the correct gateway.
70///
71/// Load balancing is achieved by tracking the number of connections
72/// (fanin) per gateway node.
73#[derive(Debug)]
74pub struct PropagationTreeRouter {
75    /// The root node of this router.
76    root: String,
77    /// Fanout setting (0 = no limit).
78    fanout: usize,
79    /// Connections count per gateway node (for load balancing).
80    nodes_fanin: HashMap<String, usize>,
81    /// Routing table: list of (network, nexthops).
82    table: Vec<RouteEntry>,
83    /// Set of nodes marked as unreachable.
84    unreachable_hosts: NodeSet,
85}
86
87impl PropagationTreeRouter {
88    /// Create a new router from a root node and topology tree.
89    ///
90    /// The routing table is generated by traversing the tree from the root
91    /// node's position downward.
92    ///
93    /// # Errors
94    /// Returns [`RouteResolvingError::InvalidRoot`] if the root node cannot
95    /// be found in the topology.
96    pub fn new(
97        root: &str,
98        topology: &TopologyTree,
99        fanout: usize,
100    ) -> Result<Self, RouteResolvingError> {
101        let mut router = Self {
102            root: root.to_string(),
103            fanout,
104            nodes_fanin: HashMap::new(),
105            table: Vec::new(),
106            unreachable_hosts: NodeSet::new(),
107        };
108        router.table_generate(root, topology)?;
109        Ok(router)
110    }
111
112    /// Generate the routing table from the topology tree.
113    ///
114    /// For each direct child group of the root, compute the set of all
115    /// transitively reachable nodes (the "network") and store it alongside
116    /// the next-hop gateway nodeset.
117    fn table_generate(
118        &mut self,
119        root: &str,
120        topology: &TopologyTree,
121    ) -> Result<(), RouteResolvingError> {
122        // Validate the root exists in the topology
123        let _root_group = topology
124            .find_nodegroup(root)
125            .map_err(|_| RouteResolvingError::InvalidRoot(root.to_string()))?;
126
127        self.table.clear();
128
129        // We need to find the root TopologyNodeGroup and enumerate its
130        // direct children. Each child group becomes a route entry, with
131        // its "network" being all transitively reachable nodes below it.
132        let root_tng = topology
133            .root()
134            .ok_or_else(|| RouteResolvingError::InvalidRoot(root.to_string()))?;
135
136        // The root of the topology tree should contain our root node.
137        // If the root_tng doesn't contain our node, we need to find
138        // the subtree rooted at our node.
139        let subtree_root = Self::find_group_containing(root_tng, root)
140            .ok_or_else(|| RouteResolvingError::InvalidRoot(root.to_string()))?;
141
142        for child in subtree_root.children() {
143            let mut network = NodeSet::new();
144            Self::collect_reachable(child, &mut network);
145            self.table.push(RouteEntry {
146                network,
147                nexthops: child.nodeset.clone(),
148            });
149        }
150
151        Ok(())
152    }
153
154    /// Find the TopologyNodeGroup that contains the given node name.
155    fn find_group_containing<'a>(
156        group: &'a crate::topology::TopologyNodeGroup,
157        node: &str,
158    ) -> Option<&'a crate::topology::TopologyNodeGroup> {
159        if group.nodeset.contains(node) {
160            return Some(group);
161        }
162        for child in group.children() {
163            if let Some(found) = Self::find_group_containing(child, node) {
164                return Some(found);
165            }
166        }
167        None
168    }
169
170    /// Recursively collect all nodes reachable below a group (its children,
171    /// grandchildren, etc.) into `dest`. Does NOT include the group itself.
172    fn collect_reachable(group: &crate::topology::TopologyNodeGroup, dest: &mut NodeSet) {
173        for child in group.children() {
174            dest.update(&child.nodeset);
175            Self::collect_reachable(child, dest);
176        }
177    }
178
179    /// Dispatch nodes from a target nodeset to directly connected gateways.
180    ///
181    /// Acts as an iterator, yielding `(gateway_nodeset, target_nodes)` pairs.
182    /// Provides load balancing between gateways. Remaining nodes that don't
183    /// match any route are treated as directly connected.
184    pub fn dispatch(&self, dst: &NodeSet) -> Vec<(NodeSet, NodeSet)> {
185        let mut remaining = dst.clone();
186        let mut results = Vec::new();
187
188        // Check for remote targets requiring a gateway
189        for entry in &self.table {
190            let dst_inter = entry.network.intersection(&remaining);
191            if !dst_inter.is_empty() {
192                remaining.difference_update(&dst_inter);
193                // For each host in the intersection, find best next hop
194                for host_name in dst_inter.iter() {
195                    if let Ok(hop) = self.next_hop_str(&host_name) {
196                        let mut hop_ns = NodeSet::new();
197                        let _ = hop_ns.update_str(&hop);
198                        let mut host_ns = NodeSet::new();
199                        let _ = host_ns.update_str(&host_name);
200                        results.push((hop_ns, host_ns));
201                    }
202                }
203            }
204        }
205
206        // Remaining nodes are directly connected
207        if !remaining.is_empty() {
208            results.push((remaining.clone(), remaining));
209        }
210
211        results
212    }

214    /// Perform next-hop resolution for a single destination node.
215    ///
216    /// Returns the best gateway to use based on current load (connections
217    /// tracked in `nodes_fanin`).
218    pub fn next_hop(&mut self, dst: &NodeSet) -> Result<NodeSet, RouteResolvingError> {
219        let dst_str = dst.to_string();
220
221        // Check unreachable
222        if !self.unreachable_hosts.intersection(dst).is_empty() {
223            return Err(RouteResolvingError::Unreachable(dst_str));
224        }
225
226        // Can't resolve if source == destination
227        if self.root == dst_str {
228            return Err(RouteResolvingError::SameSourceDest(
229                self.root.clone(),
230                dst_str,
231            ));
232        }
233
234        for entry in &self.table {
235            // destination contained in current network
236            if !entry.network.intersection(dst).is_empty() {
237                let res = self.best_next_hop(&entry.nexthops)?;
238                let res_str = res.to_string();
239                *self.nodes_fanin.entry(res_str).or_insert(0) += dst.len();
240                return Ok(res);
241            }
242            // destination is a direct next-hop
243            if !entry.nexthops.intersection(dst).is_empty() {
244                return Ok(dst.clone());
245            }
246        }
247
248        Err(RouteResolvingError::NoRoute(dst_str))
249    }
250
251    /// String-based next_hop for convenience (non-mutating for dispatch).
252    fn next_hop_str(&self, dst: &str) -> Result<String, RouteResolvingError> {
253        // Check unreachable
254        if self.unreachable_hosts.contains(dst) {
255            return Err(RouteResolvingError::Unreachable(dst.to_string()));
256        }
257
258        if self.root == dst {
259            return Err(RouteResolvingError::SameSourceDest(
260                self.root.clone(),
261                dst.to_string(),
262            ));
263        }
264
265        let dst_ns = NodeSet::parse(dst)?;
266
267        for entry in &self.table {
268            if !entry.network.intersection(&dst_ns).is_empty() {
269                let res = self.best_next_hop(&entry.nexthops)?;
270                return Ok(res.to_string());
271            }
272            if !entry.nexthops.intersection(&dst_ns).is_empty() {
273                return Ok(dst.to_string());
274            }
275        }
276
277        Err(RouteResolvingError::NoRoute(dst.to_string()))
278    }

280    /// Mark a node as unreachable.

282        let _ = self.unreachable_hosts.update_str(dst);
283    }

285    /// Find the best next-hop from a set of candidates.
286    ///
287    /// Selects the gateway with the fewest current connections (load balancing).
288    fn best_next_hop(&self, candidates: &NodeSet) -> Result<NodeSet, RouteResolvingError> {
289        let available = candidates.difference(&self.unreachable_hosts);
290        if available.is_empty() {
291            return Err(RouteResolvingError::NoRoute(
292                "all candidates unreachable".into(),
293            ));
294        }
295
296        let mut best: Option<String> = None;
297        let mut best_connections = usize::MAX;
298
299        for host in available.iter() {
300            let connections = self.nodes_fanin.get(&host).copied().unwrap_or(0);
301            if connections < best_connections {
302                best_connections = connections;
303                best = Some(host);
304            }
305        }
306
307        match best {
308            Some(host) => {
309                let mut ns = NodeSet::new();
310                let _ = ns.update_str(&host);
311                Ok(ns)
312            }
313            None => Err(RouteResolvingError::NoRoute(
314                "no candidate available".into(),
315            )),
316        }
317    }

339/// State of a PropagationChannel's finite state machine.
340#[derive(Debug, Clone, Copy, PartialEq, Eq)]
341pub enum ChannelState {
342    /// Initial state — channel not yet opened.
343    Init,
344    /// Configuration sent, waiting for ACK.
345    Config,
346    /// Control message sent, waiting for ACK.
347    Control,
348    /// Gathering results from the subtree.
349    Gather,
350    /// Channel is closed.
351    Closed,
352}
353
354/// Admin-node propagation channel.
355///
356/// Manages the lifecycle of communication with a gateway node:
357/// 1. **Init**: Send configuration (serialized topology).
358/// 2. **Config**: Wait for ACK, then send control message (command).
359/// 3. **Control**: Wait for ACK, then enter gather state.
360/// 4. **Gather**: Collect results (stdout, stderr, retcodes, timeouts, routing).
361///
362/// This is a state machine that drives the [`Channel`] transport.
363#[derive(Debug)]
364pub struct PropagationChannel {
365    /// The underlying communication channel.
366    pub channel: Channel,
367    /// The gateway node this channel connects to.
368    pub gateway: String,
369    /// Current FSM state.
370    pub state: ChannelState,
371    /// Queue of control messages waiting to be sent.
372    send_queue: VecDeque<Message>,
373    /// Last return code from the gateway process.
374    pub last_rc: Option<i32>,
375    /// Write history for tracking ACKs to write operations.
376    /// Each entry: (msgid, nodes, byte_count).
377    write_history: VecDeque<(u64, String, usize)>,
378    /// Collected results: (node, stream_name, data).
379    pub results: Vec<PropagationResult>,
380}
381
382/// A result received from the propagation subtree.
383#[derive(Debug, Clone)]
384pub enum PropagationResult {
385    /// Standard output line from a node.
386    StdOut { node: String, data: Vec<u8> },
387    /// Standard error line from a node.
388    StdErr { node: String, data: Vec<u8> },
389    /// Return code from a node.
390    Retcode { node: String, rc: i32 },
391    /// Timeout for a node.
392    Timeout { node: String },
393    /// Routing event.
394    Routing {
395        event: String,
396        gateway: String,
397        targets: String,
398    },
399}

401impl PropagationChannel {
402    /// Create a new propagation channel for a gateway.
403    pub fn new(gateway: &str) -> Self {
404        Self {
405            channel: Channel::new(true), // initiator
406            gateway: gateway.to_string(),
407            state: ChannelState::Init,
408            send_queue: VecDeque::new(),
409            last_rc: None,
410            write_history: VecDeque::new(),
411            results: Vec::new(),
412        }
413    }
414
415    /// Start the propagation channel — send configuration.
416    ///
417    /// Serializes the topology and sends it as a ConfigurationMessage.
418    pub fn start(&mut self, topology_bytes: &[u8]) {
419        // Open the channel
420        self.channel.opened = true;
421
422        // Send configuration message
423        let mut cfg = Message::configuration(&self.gateway);
424        cfg.data_encode(topology_bytes);
425        self.channel.send(&cfg);
426        self.state = ChannelState::Config;
427    }
428
429    /// Enqueue a control message for sending.
430    ///
431    /// If the channel is already set up and the queue is empty, the message
432    /// is sent immediately. Otherwise it's queued for later.
433    pub fn send_queued(&mut self, msg: Message) {
434        if self.channel.setup && self.send_queue.is_empty() {
435            self.channel.send(&msg);
436        } else {
437            self.send_queue.push_back(msg);
438        }
439    }
440
441    /// Send one queued message, if any.
442    fn send_dequeue(&mut self) {
443        if let Some(msg) = self.send_queue.pop_front() {
444            self.channel.send(&msg);
445        }
446    }
447
448    /// Create and enqueue a shell command for execution through the channel.
449    pub fn shell(&mut self, nodes: &str, command: &[u8], worker_id: u64) {
450        let mut ctl = Message::control(worker_id);
451        if let Message::Control {
452            ref mut action,
453            ref mut target,
454            ..
455        } = ctl
456        {
457            *action = "shell".to_string();
458            *target = nodes.to_string();
459        }
460        ctl.data_encode(command);
461        self.send_queued(ctl);
462    }
463
464    /// Create and enqueue a write operation through the channel.
465    pub fn write(&mut self, nodes: &str, buf: &[u8], worker_id: u64) {
466        let mut ctl = Message::control(worker_id);
467        if let Message::Control {
468            ref mut action,
469            ref mut target,
470            ..
471        } = ctl
472        {
473            *action = "write".to_string();
474            *target = nodes.to_string();
475        }
476        let msgid = ctl.msgid();
477        ctl.data_encode(buf);
478        self.write_history
479            .push_back((msgid, nodes.to_string(), buf.len()));
480        self.send_queued(ctl);
481    }

483    /// Create and enqueue an EOF operation through the channel.
484    pub fn set_write_eof(&mut self, nodes: &str, worker_id: u64) {
485        let mut ctl = Message::control(worker_id);
486        if let Message::Control {
487            ref mut action,
488            ref mut target,
489            ..
490        } = ctl
491        {
492            *action = "eof".to_string();
493            *target = nodes.to_string();
494        }
495        self.send_queued(ctl);
496    }

498    /// Process an incoming message according to the current FSM state.
499    ///
500    /// Returns any propagation results generated by this message.
501    pub fn recv(&mut self, msg: Message) -> Result<Vec<PropagationResult>, MessageProcessingError> {
502        let mut new_results = Vec::new();
503
504        match msg.ident() {
505            "END" => {
506                self.state = ChannelState::Closed;
507                return Ok(new_results);
508            }
509            "CHA" => {
510                self.channel.opened = true;
511                return Ok(new_results);
512            }
513            _ => {}
514        }
515
516        match self.state {
517            ChannelState::Config => {
518                // Waiting for ACK to configuration
519                if msg.ident() == "ACK" {
520                    self.channel.setup = true;
521                    self.state = ChannelState::Gather;
522                    self.send_dequeue();
523                }
524            }
525            ChannelState::Control | ChannelState::Gather => {
526                match msg.ident() {
527                    "ACK" => {
528                        // Check if ACK matches a write history entry
529                        if let Message::Ack { ack, .. } = &msg {
530                            if let Some(back) = self.write_history.back() {
531                                if back.0 == *ack {
532                                    let _ = self.write_history.pop_back();
533                                }
534                            }
535                        }
536                        self.send_dequeue();
537                    }
538                    "OUT" => {
539                        if let Some(nodes_str) = msg.nodes() {
540                            if let Ok(decoded) = msg.data_decode() {
541                                let ns = NodeSet::parse(nodes_str).unwrap_or_default();
542                                for node in ns.iter() {
543                                    let result = PropagationResult::StdOut {
544                                        node,
545                                        data: decoded.clone(),
546                                    };
547                                    new_results.push(result);
548                                }
549                            }
550                        }
551                    }
552                    "SER" => {
553                        if let Some(nodes_str) = msg.nodes() {
554                            if let Ok(decoded) = msg.data_decode() {
555                                let ns = NodeSet::parse(nodes_str).unwrap_or_default();
556                                for node in ns.iter() {
557                                    let result = PropagationResult::StdErr {
558                                        node,
559                                        data: decoded.clone(),
560                                    };
561                                    new_results.push(result);
562                                }
563                            }
564                        }
565                    }
566                    "RET" => {
567                        if let Message::Retcode { retcode, nodes, .. } = &msg {
568                            let ns = NodeSet::parse(nodes).unwrap_or_default();
569                            for node in ns.iter() {
570                                new_results.push(PropagationResult::Retcode { node, rc: *retcode });
571                            }
572                        }
573                    }
574                    "TIM" => {
575                        if let Some(nodes_str) = msg.nodes() {
576                            let ns = NodeSet::parse(nodes_str).unwrap_or_default();
577                            for node in ns.iter() {
578                                new_results.push(PropagationResult::Timeout { node });
579                            }
580                        }
581                    }
582                    "RTR" => {
583                        if let Message::Routing {
584                            event,
585                            gateway,
586                            targets,
587                            ..
588                        } = &msg
589                        {
590                            new_results.push(PropagationResult::Routing {
591                                event: event.clone(),
592                                gateway: gateway.clone(),
593                                targets: targets.clone(),
594                            });
595                        }
596                    }
597                    "ERR" => {
598                        if let Message::Error { reason, .. } = &msg {
599                            return Err(MessageProcessingError::ParseError(format!(
600                                "{}: {}",
601                                self.gateway, reason
602                            )));
603                        }
604                    }
605                    _ => {
606                        // Ignore unknown messages in gather state
607                    }
608                }
609            }
610            _ => {}
611        }
612
613        self.results.extend(new_results.clone());
614        Ok(new_results)
615    }

617    /// Handle the gateway process closing.
618    pub fn ev_hup(&mut self, rc: i32) {
619        self.last_rc = Some(rc);
620    }

622    /// Handle the channel closing.

640    }
