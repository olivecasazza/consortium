1//! cascade-copy — fan-out distribute a nix store path across a fleet
2//! using the cascade primitive + real `nix copy` over SSH.
3//!
4//! Usage:
5//!   cascade-copy <STORE_PATH> --inventory hosts.toml [opts]
6//!
7//! Drives a real cascade against actual hosts:
8//! - The seed host (from inventory) `nix copy`s the path to its
9//!   tree children.
10//! - As children receive the closure, they SSH-launch their own
11//!   `nix copy` to THEIR children — log-N fan-out.
12//! - Live tree visualization shows progress in real time.
13//!
14//! Differs from `cascade-viz live` (sim) in that EVERY edge is a
15//! real subprocess call against actual hosts.
16
17use std::io;
18use std::time::Duration;
19
20use anyhow::{Context, Result};
21use clap::Parser;
22use is_terminal::IsTerminal;
23
24use consortium_cli::event_render::LiveTreeRenderer;
25use consortium_cli::inventory::load_inventory;
26use consortium_cli::output::{CliOutput, OutputArgs};
27use consortium_nix::cascade::{Cascade, Log2FanOut, NetworkProfile};
28use consortium_nix::cascade_executor::NixCopyExecutor;
29use consortium_nix::cascade_strategies::{LevelTreeFanOut, MaxBottleneckSpanning, SteinerGreedy};
30
31#[derive(Parser, Debug)]
32#[command(
33    name = "cascade-copy",
34    about = "Fan-out distribute a nix store path across a fleet via cascade + real nix copy over SSH"
35)]
36struct Args {
37    /// The nix store path to distribute (e.g. /nix/store/xxx-foo-1.0).
38    /// Must already exist on the seed host.
39    store_path: String,
40
41    /// Path to the TOML inventory file with `seed` + `nodes` SSH addrs.
42    #[arg(short = 'i', long = "inventory")]
43    inventory: String,
44
45    /// Cascade strategy: level-tree (default), log2-fanout,
46    /// max-bottleneck, or steiner.
47    #[arg(short = 's', long = "strategy", default_value = "level-tree")]
48    strategy: String,
49
50    /// Fanout for level-tree (children per node, default 2).
51    #[arg(long = "fanout", default_value_t = 2)]
52    fanout: u32,
53
54    /// Per-edge `nix copy` subprocess timeout in seconds (default 300).
55    #[arg(long = "timeout", default_value_t = 300)]
56    timeout_secs: u64,
57
58    /// Cascade hard-cap rounds — if not converged by this round count,
59    /// give up. Default = 64.
60    #[arg(long = "max-rounds", default_value_t = 64)]
61    max_rounds: u32,
62
63    /// Disable live re-rendering (useful when piping or for CI).
64    #[arg(long = "no-watch")]
65    no_watch: bool,
66
67    /// Limit tree depth in the rendered output.
68    #[arg(short = 'L', long = "max-depth")]
69    max_depth: Option<usize>,
70
71    #[command(flatten)]
72    output: OutputArgs,
73}
74
75fn main() {
76    let args = Args::parse();
77    let exit_code = match run(args) {
78        Ok(c) => c,
79        Err(e) => {
80            eprintln!("cascade-copy: {e:#}");
81            1
82        }
83    };
84    std::process::exit(exit_code);
85}
86
87fn run(args: Args) -> Result<i32> {
88    let out = CliOutput::from_args(&args.output);
89
90    // Load + validate inventory.
91    let (nodes, seed, addrs) = load_inventory(&args.inventory)
92        .with_context(|| format!("loading inventory from {}", args.inventory))?;
93    if nodes.len() < 2 {
94        anyhow::bail!(
95            "inventory must have at least 1 seed + 1 target; got {} total nodes",
96            nodes.len()
97        );
98    }
99
100    out.info(format!(
101        "loaded inventory: 1 seed ({}) + {} targets",
102        nodes[0].addr,
103        nodes.len() - 1
104    ));
105    out.info(format!(
106        "store_path = {} | strategy = {} | fanout = {}",
107        args.store_path, args.strategy, args.fanout
108    ));
109
110    // Seeded set: just NodeId(0) (the seed host).
111    let mut seeded = std::collections::HashSet::new();
112    seeded.insert(seed);
113
114    // Empty network profile — no contention modeling for real
115    // deploys (the OS + nix scheduler handle it). The strategy
116    // ignores bandwidth/latency anyway for level-tree.
117    let net = NetworkProfile::default();
118
119    // The real-world executor.
120    let executor = NixCopyExecutor::new(addrs, args.store_path.clone(), seed)
121        .with_timeout(Duration::from_secs(args.timeout_secs));
122
123    // Compose nh-style header lines for the live renderer.
124    let mut header_lines: Vec<String> = Vec::new();
125    header_lines.push(format!(
126        "cascade-copy || Strategy: {} || Targets: {}",
127        args.strategy,
128        nodes.len() - 1
129    ));
130    header_lines.push(format!(
131        "Seed: {} || Path: {}",
132        nodes[0].addr, args.store_path
133    ));
134    if args.fanout != 2 {
135        header_lines.push(format!("Fanout: {}", args.fanout));
136    }
137
138    let live_eligible = io::stdout().is_terminal() && !args.no_watch;
139    let renderer = LiveTreeRenderer::new(out.color, args.max_depth).with_header_lines(header_lines);
140
141    let level_tree = LevelTreeFanOut::new(args.fanout.max(1));
142    let result = match args.strategy.as_str() {
143        "level-tree" | "level" | "tree" => Cascade::new()
144            .nodes(nodes)
145            .seeded(seeded)
146            .network(net)
147            .strategy(&level_tree)
148            .executor(&executor)
149            .max_rounds(args.max_rounds)
150            .events(if live_eligible {
151                &renderer as _
152            } else {
153                &consortium_nix::cascade_events::NullSink as _
154            })
155            .run(),
156        "log2-fanout" | "log2" => Cascade::new()
157            .nodes(nodes)
158            .seeded(seeded)
159            .network(net)
160            .strategy(&Log2FanOut)
161            .executor(&executor)
162            .max_rounds(args.max_rounds)
163            .events(if live_eligible {
164                &renderer as _
165            } else {
166                &consortium_nix::cascade_events::NullSink as _
167            })
168            .run(),
169        "max-bottleneck" | "max-bottleneck-spanning" => Cascade::new()
170            .nodes(nodes)
171            .seeded(seeded)
172            .network(net)
173            .strategy(&MaxBottleneckSpanning)
174            .executor(&executor)
175            .max_rounds(args.max_rounds)
176            .events(if live_eligible {
177                &renderer as _
178            } else {
179                &consortium_nix::cascade_events::NullSink as _
180            })
181            .run(),
182        "steiner" | "steiner-greedy" => Cascade::new()
183            .nodes(nodes)
184            .seeded(seeded)
185            .network(net)
186            .strategy(&SteinerGreedy)
187            .executor(&executor)
188            .max_rounds(args.max_rounds)
189            .events(if live_eligible {
190                &renderer as _
191            } else {
192                &consortium_nix::cascade_events::NullSink as _
193            })
194            .run(),
195        other => anyhow::bail!(
196            "unknown strategy: {other} (use level-tree, log2-fanout, max-bottleneck, or steiner)"
197        ),
198    };
199
200    let total: Duration = result.round_durations.iter().sum();
201    out.info(format!(
202        "done: {} converged / {} total in {} rounds ({:.1}s wall)",
203        result.converged.len(),
204        result.converged.len()
205            + result
206                .failed
207                .iter()
208                .flat_map(|e| e.affected_nodes())
209                .count(),
210        result.rounds,
211        total.as_secs_f64()
212    ));
213    if let Some(err) = &result.failed {
214        out.error(format!("failures: {err}"));
215        Ok(1)
216    } else {
217        Ok(0)
218    }
219}
