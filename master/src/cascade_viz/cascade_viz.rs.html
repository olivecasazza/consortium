1//! cascade-viz — visualize cascade event streams.
2//!
3//! Subcommands:
4//!   replay <TRACE_FILE>   Replay a JSONL trace file and render
5//!   live                  Run a fresh scenario and render
6//!
7//! Renderers + sinks come from `consortium_cli::event_render`. The
8//! binary is the thin shell that picks an `EventSink` (collector or
9//! JsonlWriter) for the live path, parses JSONL for the replay path,
10//! and delegates everything else to `event_render::render_events`.
11

74#[derive(Debug, Clone, Args)]
75struct LiveArgs {
76    /// Number of nodes (default 32)
77    #[arg(short = 'n', long = "nodes", default_value_t = 32)]
78    nodes: u32,
79
80    /// Strategy: level-tree (default — pre-shaped F-ary tree, each
81    /// round populates one level, matches nh's level-by-level
82    /// reveal), log2-fanout, max-bottleneck, or steiner.
83    #[arg(short = 's', long = "strategy", default_value = "level-tree")]
84    strategy: String,
85
86    /// Fanout for the level-tree strategy (children per node).
87    /// 2 → balanced binary tree, 3 → ternary, etc.
88    #[arg(long = "fanout", default_value_t = 2)]
89    fanout: u32,
90
91    /// Number of pre-seeded nodes (multiple build hosts). Round 0
92    /// will have this many parallel deploys instead of just one.
93    /// E.g. `--seeds 8` with 64 nodes = 8 parallel spinners in
94    /// round 0.
95    #[arg(long = "seeds", default_value_t = 1)]
96    seeds: u32,
97
98    /// Fraction of nodes pre-seeded (default 0.0). Overrides --seeds
99    /// when > 0.0.
100    #[arg(long = "seed-fraction", default_value_t = 0.0)]
101    seed_fraction: f64,
102
103    /// Closure size in MB (default 50)
104    #[arg(long = "closure-mb", default_value_t = 50)]
105    closure_mb: u64,
106
107    /// Bandwidth style: uniform | bimodal (default uniform)
108    #[arg(long = "bandwidth", default_value = "uniform")]
109    bandwidth: String,
110
111    /// Per-node uplink in bytes/sec (default unset = no contention)
112    #[arg(long = "uplinks")]
113    uplinks: Option<u64>,
114
115    /// RNG seed (default 0)
116    #[arg(long = "seed", default_value_t = 0)]
117    seed: u64,
118
119    /// Random per-edge failure rate (0.0 to 1.0). Each edge fails with
120    /// this probability; the outcome is deterministic from
121    /// (--failure-seed, round, src, tgt) so a given seed reproduces
122    /// the exact same failure pattern. Use to demo the cascade's
123    /// orphan re-routing — when a tree-parent fails, level-tree walks
124    /// up the heap to find an alive ancestor.
125    #[arg(long = "failure-rate", default_value_t = 0.0)]
126    failure_rate: f64,
127
128    /// Seed for the random failure RNG. Same seed + same scenario
129    /// reproduces the exact same fail/succeed sequence.
130    #[arg(long = "failure-seed", default_value_t = 0)]
131    failure_seed: u64,
132
133    /// Disable live re-rendering (collect all events first, render once
134    /// at end). Live mode is the default when stdout is a TTY + tree
135    /// format; pipes always batch since ANSI cursor codes don't make
136    /// sense in a captured stream.
137    #[arg(long = "no-watch")]
138    no_watch: bool,
139
140    /// Inject artificial wall-time delay between rounds, in milliseconds
141    /// (e.g. `--per-round-delay 200`). The deterministic sim runs in
142    /// microseconds — without this, the live re-render fires faster
143    /// than humans can perceive. Use for demos / visual debugging only;
144    /// it does NOT affect the `round_durations` reported in traces (those
145    /// stay sim-time).
146    #[arg(long = "per-round-delay")]
147    per_round_delay_ms: Option<u64>,
148}

154fn resolve_format(
155    format_str: &str,
156    no_color: bool,
157    max_depth: Option<usize>,
158) -> Result<OutputFormat> {
159    let mut fmt = OutputFormat::parse(format_str).map_err(|e| anyhow::anyhow!(e))?;
160    if let OutputFormat::Tree {
161        ref mut color,
162        max_depth: ref mut fmt_max_depth,
163    } = fmt
164    {
165        let is_tty = io::stdout().is_terminal();
166        *color = is_tty && !no_color;
167        if let Some(d) = max_depth {
168            *fmt_max_depth = Some(d);
169        }
170    }
171    Ok(fmt)
172}
173
174/// Render an in-memory event slice using either the requested format
175/// (delegated to `event_render::render_events`) or stream JSONL one
176/// event per line if the format is "jsonl".
177fn print_events(events: &[CascadeEvent], cli: &Cli) -> Result<()> {
178    if cli.format == "jsonl" {
179        for ev in events {
180            println!("{}", serde_json::to_string(ev)?);
181        }
182        return Ok(());
183    }
184    let fmt = resolve_format(&cli.format, cli.no_color, cli.max_depth)?;
185    print!("{}", render_events(events, &fmt));
186    Ok(())
187}

193fn run_replay(args: &ReplayArgs, cli: &Cli) -> Result<()> {
194    let file = File::open(&args.trace_file)
195        .with_context(|| format!("failed to open trace file: {}", args.trace_file))?;
196    let reader = BufReader::new(file);
197    let mut events: Vec<CascadeEvent> = Vec::new();
198    for (i, line) in reader.lines().enumerate() {
199        let line = line.with_context(|| format!("failed to read line {}", i + 1))?;
200        if line.trim().is_empty() {
201            continue;
202        }
203        let ev: CascadeEvent = serde_json::from_str(&line)
204            .with_context(|| format!("invalid event JSON on line {}: {line}", i + 1))?;
205        events.push(ev);
206    }
207    print_events(&events, cli)
208}

210fn run_live(args: &LiveArgs, cli: &Cli) -> Result<()> {
211    let bandwidth = match args.bandwidth.as_str() {
212        "bimodal" => BandwidthDistribution::Bimodal {
213            slow: 10 * 1024 * 1024,
214            fast: 1024 * 1024 * 1024,
215            fast_fraction: 0.3,
216        },
217        _ => BandwidthDistribution::Uniform(100 * 1024 * 1024),
218    };
219    let uplinks = args.uplinks.map(UplinkDistribution::Uniform);
220    let closure_bytes = args.closure_mb * 1024 * 1024;
221
222    // jsonl: stream live to stdout via JsonlWriter sink — events appear
223    // as they're emitted, no buffering through a Vec.
224    if cli.format == "jsonl" {
225        let sink = JsonlWriter::new(Box::new(io::stdout()));
226        run_scenario(args, closure_bytes, bandwidth, uplinks, &sink, None);
227        return Ok(());
228    }
229
230    // Live tree re-render is the default when:
231    // - format is `tree` (the only format that has a tree to redraw)
232    // - stdout is a TTY (ANSI escapes need a real terminal)
233    // - --no-watch wasn't passed
234    // Otherwise fall through to batch: collect all events, render once.
235    let live_eligible = cli.format == "tree" && io::stdout().is_terminal() && !args.no_watch;
236    if live_eligible {
237        let color = !cli.no_color;
238        // Compose nom-style header lines. Identity (line 1) always
239        // shows. Scenario knobs only appear when non-default; failure
240        // / timing config only appears when explicitly enabled.
241        // Keeps the default invocation tight, expands only with what
242        // matters.
243        let mut header_lines: Vec<String> = Vec::new();
244
245        // Line 1: identity — always present.
246        header_lines.push(format!(
247            "Strategy: {} || Nodes: {}",
248            args.strategy, args.nodes
249        ));
250
251        // Line 2: scenario knobs that differ from defaults.
252        let mut scenario: Vec<String> = Vec::new();
253        if args.fanout != 2 {
254            scenario.push(format!("Fanout: {}", args.fanout));
255        }
256        if args.seeds != 1 {
257            scenario.push(format!("Seeds: {}", args.seeds));
258        }
259        if args.closure_mb != 50 {
260            scenario.push(format!("Closure: {}MB", args.closure_mb));
261        }
262        if args.bandwidth != "uniform" {
263            scenario.push(format!("Bandwidth: {}", args.bandwidth));
264        }
265        if let Some(uplink) = args.uplinks {
266            scenario.push(format!("Uplinks: {}B/s", uplink));
267        }
268        if !scenario.is_empty() {
269            header_lines.push(scenario.join(" || "));
270        }
271
272        // Line 3: failure / timing — only when enabled.
273        let mut runtime: Vec<String> = Vec::new();
274        if args.failure_rate > 0.0 {
275            runtime.push(format!(
276                "Failures: {:.0}% (seed={})",
277                args.failure_rate * 100.0,
278                args.failure_seed
279            ));
280        }
281        if let Some(delay_ms) = args.per_round_delay_ms {
282            runtime.push(format!("Delay: {}ms/round", delay_ms));
283        }
284        if !runtime.is_empty() {
285            header_lines.push(runtime.join(" || "));
286        }
287
288        let renderer = LiveTreeRenderer::new(color, cli.max_depth).with_header_lines(header_lines);
289        let delay = args
290            .per_round_delay_ms
291            .map(std::time::Duration::from_millis);
292        run_scenario(args, closure_bytes, bandwidth, uplinks, &renderer, delay);
293        // The renderer prints the final frame on `Finished`; nothing more
294        // for us to flush.
295        return Ok(());
296    }
297
298    // Batch path: accumulate, then delegate to print_events.
299    let collector = EventCollector::new();
300    run_scenario(args, closure_bytes, bandwidth, uplinks, &collector, None);
301    let events = collector.events();
302    print_events(&events, cli)
303}

305/// Build a Cascade from `args` + run it through the given `EventSink`.
306/// All scenario wiring (nodes, seeded set, network, executor, strategy)
307/// lives here; the sink is the only consumer-specific bit.
308fn run_scenario<S: EventSink>(
309    args: &LiveArgs,
310    closure_bytes: u64,
311    bandwidth: BandwidthDistribution,
312    uplinks: Option<UplinkDistribution>,
313    sink: &S,
314    per_round_delay: Option<std::time::Duration>,
315) {
316    let n_nodes = args.nodes;
317
318    let mut alloc = NodeIdAlloc::new();
319    let nodes: Vec<CascadeNode> = (0..n_nodes)
320        .map(|_| {
321            let id = alloc.alloc();
322            CascadeNode::new(id, format!("user@host-{}", id.0))
323        })
324        .collect();
325
326    let seeded: HashSet<NodeId> = if args.seed_fraction > 0.0 {
327        let mut rng = rng_from_seed(args.seed);
328        SeedDistribution::Random {
329            fraction: args.seed_fraction,
330        }
331        .sample(&mut rng, n_nodes)
332    } else {
333        // Multi-seed: NodeIds 0..seeds. Round 0 has `seeds` parallel
334        // deploys. Capped to n_nodes, minimum 1.
335        let count = args.seeds.min(n_nodes).max(1);
336        (0..count).map(NodeId).collect()
337    };
338
339    let net: NetworkProfile = {
340        let mut rng = rng_from_seed(args.seed);
341        let mut profile = NetworkProfile::default();
342        bandwidth.populate(&mut rng, &mut profile, n_nodes);
343        if let Some(ref up) = uplinks {
344            up.populate(&mut rng, &mut profile, n_nodes);
345        }
346        profile
347    };
348
349    let failures = if args.failure_rate > 0.0 {
350        FailureSchedule::Random {
351            fraction: args.failure_rate,
352            seed: args.failure_seed,
353        }
354    } else {
355        FailureSchedule::None
356    };
357    let base_exec = consortium_fanout_sim::DeterministicExecutor::new(closure_bytes, failures);
358
359    // If --per-round-delay set, wrap the deterministic executor in a
360    // DelayingExecutor so each dispatch() sleeps for the configured
361    // duration. The sleep happens BETWEEN PlanComputed (renderer paints
362    // ⏵ spinners) and EdgeCompleted (renderer paints ✔), so the spinner
363    // frame is actually visible.
364    let delayed_exec = per_round_delay.map(|delay| DelayingExecutor {
365        inner: &base_exec as &dyn consortium_nix::cascade::RoundExecutor,
366        delay,
367    });
368    let exec: &dyn consortium_nix::cascade::RoundExecutor = delayed_exec
369        .as_ref()
370        .map(|d| d as &dyn consortium_nix::cascade::RoundExecutor)
371        .unwrap_or(&base_exec);
372
373    let level_tree = consortium_nix::cascade_strategies::LevelTreeFanOut::new(args.fanout.max(1));
374    match args.strategy.as_str() {
375        "level-tree" | "level" | "tree" => {
376            Cascade::new()
377                .nodes(nodes)
378                .seeded(seeded)
379                .network(net)
380                .strategy(&level_tree)
381                .executor(exec)
382                .events(sink)
383                .run();
384        }
385        "max-bottleneck" | "max-bottleneck-spanning" => {
386            Cascade::new()
387                .nodes(nodes)
388                .seeded(seeded)
389                .network(net)
390                .strategy(&MaxBottleneckSpanning)
391                .executor(exec)
392                .events(sink)
393                .run();
394        }
395        "steiner" | "steiner-greedy" => {
396            Cascade::new()
397                .nodes(nodes)
398                .seeded(seeded)
399                .network(net)
400                .strategy(&SteinerGreedy)
401                .executor(exec)
402                .events(sink)
403                .run();
404        }
405        _ => {
406            Cascade::new()
407                .nodes(nodes)
408                .seeded(seeded)
409                .network(net)
410                .strategy(&Log2FanOut)
411                .executor(exec)
412                .events(sink)
413                .run();
414        }
415    }
416}
