1//! Tree propagation engine.
2//!
3//! Rust implementation of `ClusterShell.Propagation`.
4//!
5//! This module provides:
6//! - [`PropagationTreeRouter`]: Routes commands through the topology tree by
7//!   computing next-hop gateways and managing fanout/load-balancing.
8//! - [`PropagationChannel`]: State machine for gateway-to-gateway channel
9//!   lifecycle (configuration, control, gather phases).

23/// Errors raised during route resolving operations.
24#[derive(Error, Debug)]
25pub enum RouteResolvingError {
26    /// Invalid root or gateway node.
27    #[error("Invalid root or gateway node: {0}")]
28    InvalidRoot(String),
29
30    /// No route available to the destination.
31    #[error("No route available to {0}")]
32    NoRoute(String),
33
34    /// Destination host is unreachable.
35    #[error("Invalid destination: {0}, host is unreachable")]
36    Unreachable(String),
37
38    /// Source equals destination.
39    #[error("Invalid resolution request: {0} -> {1}")]
40    SameSourceDest(String, String),
41
42    /// Topology error.
43    #[error("Topology error: {0}")]
44    Topology(#[from] TopologyError),
45
46    /// NodeSet error.
47    #[error("NodeSet error: {0}")]
48    NodeSet(#[from] crate::node_set::NodeSetError),
49}
50
51/// Errors raised by [`PropagationChannel::recv`].
52#[derive(Error, Debug)]
53pub enum RecvError {
54    /// Protocol/message-level failure (invalid payload, parse errors, ...).
55    #[error("{0}")]
56    Message(#[from] MessageProcessingError),
57
58    /// Gateway runtime error received after channel setup. Upstream raises
59    /// `TopologyError("%s: %s" % (gateway, reason))` here
60    /// (PropagationChannel.recv_ctl); the rendered text is kept identical,
61    /// e.g. "gw1: bad news".
62    #[error("{0}")]
63    Gateway(String),
64}

70/// A routing table entry: (reachable network, next-hop gateways).
71#[derive(Debug, Clone)]
72pub struct RouteEntry {
73    /// Set of all nodes reachable through this route (transitive children).
74    network: NodeSet,
75    /// Direct next-hop gateway(s) for this route.
76    nexthops: NodeSet,
77}

79/// Performs route resolving operations within a propagation tree.
80///
81/// The router parses the topology tree to generate a routing table, then
82/// provides [`next_hop()`](PropagationTreeRouter::next_hop) and
83/// [`dispatch()`](PropagationTreeRouter::dispatch) methods for directing
84/// messages to the correct gateway.
85///
86/// Load balancing is achieved by tracking the number of connections
87/// (fanin) per gateway node.
88#[derive(Debug)]
89pub struct PropagationTreeRouter {
90    /// The root node of this router.
91    root: String,
92    /// Fanout setting (0 = no limit).
93    fanout: usize,
94    /// Connections count per gateway node (for load balancing).
95    nodes_fanin: HashMap<String, usize>,
96    /// Routing table: list of (network, nexthops).
97    table: Vec<RouteEntry>,
98    /// Set of nodes marked as unreachable.
99    unreachable_hosts: NodeSet,
100}
101
102impl PropagationTreeRouter {
103    /// Create a new router from a root node and topology tree.
104    ///
105    /// The routing table is generated by traversing the tree from the root
106    /// node's position downward.
107    ///
108    /// # Errors
109    /// Returns [`RouteResolvingError::InvalidRoot`] if the root node cannot
110    /// be found in the topology.
111    pub fn new(
112        root: &str,
113        topology: &TopologyTree,
114        fanout: usize,
115    ) -> Result<Self, RouteResolvingError> {
116        let mut router = Self {
117            root: root.to_string(),
118            fanout,
119            nodes_fanin: HashMap::new(),
120            table: Vec::new(),
121            unreachable_hosts: NodeSet::new(),
122        };
123        router.table_generate(root, topology)?;
124        Ok(router)
125    }
126
127    /// Generate the routing table from the topology tree.
128    ///
129    /// For each direct child group of the root, compute the set of all
130    /// transitively reachable nodes (the "network") and store it alongside
131    /// the next-hop gateway nodeset.
132    fn table_generate(
133        &mut self,
134        root: &str,
135        topology: &TopologyTree,
136    ) -> Result<(), RouteResolvingError> {
137        // Validate the root exists in the topology
138        let _root_group = topology
139            .find_nodegroup(root)
140            .map_err(|_| RouteResolvingError::InvalidRoot(root.to_string()))?;
141
142        self.table.clear();
143
144        // We need to find the root TopologyNodeGroup and enumerate its
145        // direct children. Each child group becomes a route entry, with
146        // its "network" being all transitively reachable nodes below it.
147        let root_tng = topology
148            .root()
149            .ok_or_else(|| RouteResolvingError::InvalidRoot(root.to_string()))?;
150
151        // The root of the topology tree should contain our root node.
152        // If the root_tng doesn't contain our node, we need to find
153        // the subtree rooted at our node.
154        let subtree_root = Self::find_group_containing(root_tng, root)
155            .ok_or_else(|| RouteResolvingError::InvalidRoot(root.to_string()))?;
156
157        for child in subtree_root.children() {
158            let mut network = NodeSet::new();
159            Self::collect_reachable(child, &mut network);
160            self.table.push(RouteEntry {
161                network,
162                nexthops: child.nodeset.clone(),
163            });
164        }
165
166        Ok(())
167    }
168
169    /// Find the TopologyNodeGroup that contains the given node name.
170    fn find_group_containing<'a>(
171        group: &'a crate::topology::TopologyNodeGroup,
172        node: &str,
173    ) -> Option<&'a crate::topology::TopologyNodeGroup> {
174        if group.nodeset.contains(node) {
175            return Some(group);
176        }
177        for child in group.children() {
178            if let Some(found) = Self::find_group_containing(child, node) {
179                return Some(found);
180            }
181        }
182        None
183    }
184
185    /// Recursively collect all nodes reachable below a group (its children,
186    /// grandchildren, etc.) into `dest`. Does NOT include the group itself.
187    fn collect_reachable(group: &crate::topology::TopologyNodeGroup, dest: &mut NodeSet) {
188        for child in group.children() {
189            dest.update(&child.nodeset);
190            Self::collect_reachable(child, dest);
191        }
192    }
193
194    /// Dispatch nodes from a target nodeset to directly connected gateways.
195    ///
196    /// Acts as an iterator, yielding `(gateway_nodeset, target_nodes)` pairs.
197    /// Provides load balancing between gateways. Remaining nodes that don't
198    /// match any route are treated as directly connected.
199    pub fn dispatch(&self, dst: &NodeSet) -> Vec<(NodeSet, NodeSet)> {
200        let mut remaining = dst.clone();
201        let mut results = Vec::new();
202
203        // Check for remote targets requiring a gateway
204        for entry in &self.table {
205            let dst_inter = entry.network.intersection(&remaining);
206            if !dst_inter.is_empty() {
207                remaining.difference_update(&dst_inter);
208                // For each host in the intersection, find best next hop
209                for host_name in dst_inter.iter() {
210                    if let Ok(hop) = self.next_hop_str(&host_name) {
211                        let mut hop_ns = NodeSet::new();
212                        let _ = hop_ns.update_str(&hop);
213                        let mut host_ns = NodeSet::new();
214                        let _ = host_ns.update_str(&host_name);
215                        results.push((hop_ns, host_ns));
216                    }
217                }
218            }
219        }
220
221        // Remaining nodes are directly connected
222        if !remaining.is_empty() {
223            results.push((remaining.clone(), remaining));
224        }
225
226        results
227    }

229    /// Perform next-hop resolution for a single destination node.
230    ///
231    /// Returns the best gateway to use based on current load (connections
232    /// tracked in `nodes_fanin`).
233    pub fn next_hop(&mut self, dst: &NodeSet) -> Result<NodeSet, RouteResolvingError> {
234        let dst_str = dst.to_string();
235
236        // Check unreachable
237        if !self.unreachable_hosts.intersection(dst).is_empty() {
238            return Err(RouteResolvingError::Unreachable(dst_str));
239        }
240
241        // Can't resolve if source == destination
242        if self.root == dst_str {
243            return Err(RouteResolvingError::SameSourceDest(
244                self.root.clone(),
245                dst_str,
246            ));
247        }
248
249        for entry in &self.table {
250            // destination contained in current network
251            if !entry.network.intersection(dst).is_empty() {
252                let res = self.best_next_hop(&entry.nexthops)?;
253                let res_str = res.to_string();
254                *self.nodes_fanin.entry(res_str).or_insert(0) += dst.len();
255                return Ok(res);
256            }
257            // destination is a direct next-hop
258            if !entry.nexthops.intersection(dst).is_empty() {
259                return Ok(dst.clone());
260            }
261        }
262
263        Err(RouteResolvingError::NoRoute(dst_str))
264    }
265
266    /// String-based next_hop for convenience (non-mutating for dispatch).
267    fn next_hop_str(&self, dst: &str) -> Result<String, RouteResolvingError> {
268        // Check unreachable
269        if self.unreachable_hosts.contains(dst) {
270            return Err(RouteResolvingError::Unreachable(dst.to_string()));
271        }
272
273        if self.root == dst {
274            return Err(RouteResolvingError::SameSourceDest(
275                self.root.clone(),
276                dst.to_string(),
277            ));
278        }
279
280        let dst_ns = NodeSet::parse(dst)?;
281
282        for entry in &self.table {
283            if !entry.network.intersection(&dst_ns).is_empty() {
284                let res = self.best_next_hop(&entry.nexthops)?;
285                return Ok(res.to_string());
286            }
287            if !entry.nexthops.intersection(&dst_ns).is_empty() {
288                return Ok(dst.to_string());
289            }
290        }
291
292        Err(RouteResolvingError::NoRoute(dst.to_string()))
293    }

295    /// Mark a node as unreachable.
296    pub fn mark_unreachable(&mut self, dst: &str) {
297        let _ = self.unreachable_hosts.update_str(dst);
298    }

300    /// Find the best next-hop from a set of candidates.
301    ///
302    /// Selects the gateway with the fewest current connections (load balancing).
303    fn best_next_hop(&self, candidates: &NodeSet) -> Result<NodeSet, RouteResolvingError> {
304        let available = candidates.difference(&self.unreachable_hosts);
305        if available.is_empty() {
306            return Err(RouteResolvingError::NoRoute(
307                "all candidates unreachable".into(),
308            ));
309        }
310
311        let mut best: Option<String> = None;
312        let mut best_connections = usize::MAX;
313
314        for host in available.iter() {
315            let connections = self.nodes_fanin.get(&host).copied().unwrap_or(0);
316            if connections < best_connections {
317                best_connections = connections;
318                best = Some(host);
319            }
320        }
321
322        match best {
323            Some(host) => {
324                let mut ns = NodeSet::new();
325                let _ = ns.update_str(&host);
326                Ok(ns)
327            }
328            None => Err(RouteResolvingError::NoRoute(
329                "no candidate available".into(),
330            )),
331        }
332    }

354/// State of a PropagationChannel's finite state machine.
355#[derive(Debug, Clone, Copy, PartialEq, Eq)]
356pub enum ChannelState {
357    /// Initial state — channel not yet opened.
358    Init,
359    /// Configuration sent, waiting for ACK.
360    Config,
361    /// Control message sent, waiting for ACK.
362    Control,
363    /// Gathering results from the subtree.
364    Gather,
365    /// Channel is closed.
366    Closed,
367}
368
369/// Admin-node propagation channel.
370///
371/// Manages the lifecycle of communication with a gateway node:
372/// 1. **Init**: Send configuration (serialized topology).
373/// 2. **Config**: Wait for ACK, then send control message (command).
374/// 3. **Control**: Wait for ACK, then enter gather state.
375/// 4. **Gather**: Collect results (stdout, stderr, retcodes, timeouts, routing).
376///
377/// This is a state machine that drives the [`Channel`] transport.
378#[derive(Debug)]
379pub struct PropagationChannel {
380    /// The underlying communication channel.
381    pub channel: Channel,
382    /// The gateway node this channel connects to.
383    pub gateway: String,
384    /// Current FSM state.
385    pub state: ChannelState,
386    /// Queue of control messages waiting to be sent, with the target nodes
387    /// whose pickup event must fire once the message is actually sent.
388    /// Mirrors upstream's `self._sendq` of (ctl, pickup_worker, pickup_nodes).
389    send_queue: VecDeque<(Message, Vec<String>)>,
390    /// Pickup notifications produced when a CTL actually goes on the wire
391    /// (upstream #594: `ev_pickup` fires only when the command really left
392    /// the local process). Drained by the caller (TreeWorker), which
393    /// forwards them as `ev_pickup`. write()/eof CTLs never add entries.
394    pub pickup_events: Vec<String>,
395    /// Last return code from the gateway process.
396    pub last_rc: Option<i32>,
397    /// Write history for tracking ACKs to write operations.
398    /// Each entry: (msgid, nodes, byte_count).
399    write_history: VecDeque<(u64, String, usize)>,
400    /// Collected results: (node, stream_name, data).
401    pub results: Vec<PropagationResult>,
402}

404/// A result received from the propagation subtree.
405#[derive(Debug, Clone)]
406pub enum PropagationResult {
407    /// Standard output line from a node.
408    StdOut { node: String, data: Vec<u8> },
409    /// Standard error line from a node.
410    StdErr { node: String, data: Vec<u8> },
411    /// Return code from a node.
412    Retcode { node: String, rc: i32 },
413    /// Timeout for a node.
414    Timeout { node: String },
415    /// Routing event.
416    Routing {
417        event: String,
418        gateway: String,
419        targets: String,
420    },
421}

423/// Split a byte buffer into lines, mirroring Python's `bytes.splitlines()`
424/// for the ASCII line boundaries used by the tree protocol (\n, \r\n, \r).
425/// Lines are returned without their terminator; a trailing partial line is
426/// kept. Matches the oracle's per-line stderr/stdout reporting.
427fn split_lines(buf: &[u8]) -> Vec<Vec<u8>> {
428    let mut lines = Vec::new();
429    let mut start = 0;
430    let mut i = 0;
431    while i < buf.len() {
432        if buf[i] == b'\n' || buf[i] == b'\r' {
433            lines.push(buf[start..i].to_vec());
434            if buf[i] == b'\r' && i + 1 < buf.len() && buf[i + 1] == b'\n' {
435                i += 1;
436            }
437            i += 1;
438            start = i;
439        } else {
440            i += 1;
441        }
442    }
443    if start < buf.len() {
444        lines.push(buf[start..].to_vec());
445    }
446    lines
447}

449impl PropagationChannel {
450    /// Create a new propagation channel for a gateway.
451    pub fn new(gateway: &str) -> Self {
452        Self {
453            channel: Channel::new(true), // initiator
454            gateway: gateway.to_string(),
455            state: ChannelState::Init,
456            send_queue: VecDeque::new(),
457            pickup_events: Vec::new(),
458            last_rc: None,
459            write_history: VecDeque::new(),
460            results: Vec::new(),
461        }
462    }
463
464    /// Start the propagation channel — send configuration.
465    ///
466    /// Serializes the topology and sends it as a ConfigurationMessage.
467    pub fn start(&mut self, topology_bytes: &[u8]) {
468        // Open the channel
469        self.channel.opened = true;
470
471        // Send configuration message
472        let mut cfg = Message::configuration(&self.gateway);
473        cfg.data_encode(topology_bytes);
474        self.channel.send(&cfg);
475        self.state = ChannelState::Config;
476    }
477
478    /// Actually push a CTL message on the wire.
479    ///
480    /// Mirrors upstream `PropagationChannel._send_ctl`: if `pickup_nodes` is
481    /// non-empty, a pickup notification is recorded for every target node
482    /// AFTER the send — so a pickup event only fires when the command really
483    /// left the local process (#594). write()/set_write_eof() CTLs never
484    /// carry pickup nodes.
485    fn send_ctl(&mut self, ctl: Message, pickup_nodes: Vec<String>) {
486        self.channel.send(&ctl);
487        self.pickup_events.extend(pickup_nodes);
488    }
489
490    /// Enqueue a control message for sending.
491    ///
492    /// If the channel is already set up and the queue is empty, the message
493    /// is sent immediately. Otherwise it's queued for later.
494    pub fn send_queued(&mut self, msg: Message) {
495        self.send_queued_pickup(msg, Vec::new());
496    }
497
498    /// Helper used to send a message, using the msg queue if needed, with
499    /// optional pickup nodes (mirrors upstream `send_queued(ctl,
500    /// pickup_worker=..., pickup_nodes=...)`).
501    fn send_queued_pickup(&mut self, msg: Message, pickup_nodes: Vec<String>) {
502        if self.channel.setup && self.send_queue.is_empty() {
503            // send now if channel is setup and sendq empty
504            self.send_ctl(msg, pickup_nodes);
505        } else {
506            self.send_queue.push_back((msg, pickup_nodes));
507        }
508    }
509
510    /// Send one queued message, if any (mirrors upstream `send_dequeue`:
511    /// pickup notifications fire here too, at actual send time).
512    fn send_dequeue(&mut self) {
513        if let Some((msg, pickup_nodes)) = self.send_queue.pop_front() {
514            self.send_ctl(msg, pickup_nodes);
515        }
516    }
517
518    /// Create and enqueue a shell command for execution through the channel.
519    ///
520    /// Mirrors upstream `PropagationChannel.shell`: the target nodes are
521    /// registered for pickup notification, which fires only when the CTL is
522    /// actually written on the wire (#594).
523    pub fn shell(&mut self, nodes: &str, command: &[u8], worker_id: u64) {
524        let mut ctl = Message::control(worker_id);
525        if let Message::Control {
526            ref mut action,
527            ref mut target,
528            ..
529        } = ctl
530        {
531            *action = "shell".to_string();
532            *target = nodes.to_string();
533        }
534        ctl.data_encode(command);
535        // Pass nodes so the pickup notification fires only when the CTL
536        // actually leaves the local process (upstream #594).
537        let pickup_nodes = NodeSet::parse(nodes)
538            .map(|ns| ns.iter().collect())
539            .unwrap_or_else(|_| vec![nodes.to_string()]);
540        self.send_queued_pickup(ctl, pickup_nodes);
541    }
542
543    /// Create and enqueue a write operation through the channel.
544    pub fn write(&mut self, nodes: &str, buf: &[u8], worker_id: u64) {
545        let mut ctl = Message::control(worker_id);
546        if let Message::Control {
547            ref mut action,
548            ref mut target,
549            ..
550        } = ctl
551        {
552            *action = "write".to_string();
553            *target = nodes.to_string();
554        }
555        let msgid = ctl.msgid();
556        ctl.data_encode(buf);
557        self.write_history
558            .push_back((msgid, nodes.to_string(), buf.len()));
559        self.send_queued(ctl);
560    }
561
562    /// Create and enqueue an EOF operation through the channel.
563    pub fn set_write_eof(&mut self, nodes: &str, worker_id: u64) {
564        let mut ctl = Message::control(worker_id);
565        if let Message::Control {
566            ref mut action,
567            ref mut target,
568            ..
569        } = ctl
570        {
571            *action = "eof".to_string();
572            *target = nodes.to_string();
573        }
574        self.send_queued(ctl);
575    }

577    /// Report `buf` as stderr of `nodeset`, as seen from the gateway.
578    ///
579    /// Mirrors upstream `PropagationChannel._report_stderr`: a trailing
580    /// newline is appended before splitting so that an empty buffer still
581    /// yields one (empty) line (#249), and each line is reported separately
582    /// for every node of the nodeset.
583    fn report_stderr(&self, nodeset: &NodeSet, buf: &[u8]) -> Vec<PropagationResult> {
584        let mut terminated = buf.to_vec();
585        terminated.push(b'\n');
586        let mut results = Vec::new();
587        for line in split_lines(&terminated) {
588            for node in nodeset.iter() {
589                results.push(PropagationResult::StdErr {
590                    node,
591                    data: line.clone(),
592                });
593            }
594        }
595        results
596    }

598    /// Process an incoming message according to the current FSM state.
599    ///
600    /// Returns any propagation results generated by this message.
601    pub fn recv(&mut self, msg: Message) -> Result<Vec<PropagationResult>, RecvError> {
602        let mut new_results = Vec::new();
603
604        // EndMessage: close the channel (upstream: got EndMessage; closing)
605        if msg.ident() == "END" {
606            self.state = ChannelState::Closed;
607            self.channel.close();
608            return Ok(new_results);
609        }
610
611        // Non-routed stderr (srcid == 0): error messages received when the
612        // channel is not established yet, or gateway-related messages, are
613        // reported as stderr of the given nodes in ANY state — upstream
614        // handles this before the setup dispatch (90d3195-range).
615        if msg.ident() == "SER" && msg.srcid() == Some(0) {
616            let decoded = msg.data_decode()?;
617            let nodeset = NodeSet::parse(msg.nodes().unwrap_or_default()).unwrap_or_default();
618            let results = self.report_stderr(&nodeset, &decoded);
619            self.results.extend(results.clone());
620            return Ok(results);
621        }
622
623        // StartMessage: gateway greeting opens the channel
624        if msg.ident() == "CHA" {
625            self.channel.opened = true;
626            return Ok(new_results);
627        }
628
629        match self.state {
630            ChannelState::Config => {
631                // Waiting for ACK to configuration
632                match msg.ident() {
633                    "ACK" => {
634                        self.channel.setup = true;
635                        self.state = ChannelState::Gather;
636                        self.send_dequeue();
637                    }
638                    "ERR" => {
639                        // Gateway error before setup (eg. invalid CFG
640                        // payload): report the reason as gateway stderr
641                        // instead of dropping it (upstream 90d3195).
642                        if let Message::Error { reason, .. } = &msg {
643                            let gw = NodeSet::parse(&self.gateway).unwrap_or_default();
644                            let results = self.report_stderr(&gw, reason.as_bytes());
645                            self.results.extend(results.clone());
646                            return Ok(results);
647                        }
648                    }
649                    _ => {}
650                }
651            }
652            ChannelState::Control | ChannelState::Gather => {
653                match msg.ident() {
654                    "ACK" => {
655                        // Check if ACK matches a write history entry
656                        if let Message::Ack { ack, .. } = &msg {
657                            if let Some(back) = self.write_history.back() {
658                                if back.0 == *ack {
659                                    let _ = self.write_history.pop_back();
660                                }
661                            }
662                        }
663                        self.send_dequeue();
664                    }
665                    "OUT" => {
666                        if let Some(nodes_str) = msg.nodes() {
667                            if let Ok(decoded) = msg.data_decode() {
668                                let ns = NodeSet::parse(nodes_str).unwrap_or_default();
669                                for node in ns.iter() {
670                                    let result = PropagationResult::StdOut {
671                                        node,
672                                        data: decoded.clone(),
673                                    };
674                                    new_results.push(result);
675                                }
676                            }
677                        }
678                    }
679                    "SER" => {
680                        // Routed stderr: reported per line, like upstream
681                        // recv_ctl's `decoded + b'\n'` splitlines() loop.
682                        if let Some(nodes_str) = msg.nodes() {
683                            if let Ok(decoded) = msg.data_decode() {
684                                let ns = NodeSet::parse(nodes_str).unwrap_or_default();
685                                new_results.extend(self.report_stderr(&ns, &decoded));
686                            }
687                        }
688                    }
689                    "RET" => {
690                        if let Message::Retcode { retcode, nodes, .. } = &msg {
691                            let ns = NodeSet::parse(nodes).unwrap_or_default();
692                            for node in ns.iter() {
693                                new_results.push(PropagationResult::Retcode { node, rc: *retcode });
694                            }
695                        }
696                    }
697                    "TIM" => {
698                        if let Some(nodes_str) = msg.nodes() {
699                            let ns = NodeSet::parse(nodes_str).unwrap_or_default();
700                            for node in ns.iter() {
701                                new_results.push(PropagationResult::Timeout { node });
702                            }
703                        }
704                    }
705                    "RTR" => {
706                        if let Message::Routing {
707                            event,
708                            gateway,
709                            targets,
710                            ..
711                        } = &msg
712                        {
713                            new_results.push(PropagationResult::Routing {
714                                event: event.clone(),
715                                gateway: gateway.clone(),
716                                targets: targets.clone(),
717                            });
718                        }
719                    }
720                    "ERR" => {
721                        // Tree runtime error after setup: fatal to the
722                        // channel (upstream raises TopologyError(
723                        // "%s: %s" % (gateway, reason))).
724                        if let Message::Error { reason, .. } = &msg {
725                            self.channel.close();
726                            self.state = ChannelState::Closed;
727                            return Err(RecvError::Gateway(format!(
728                                "{}: {}",
729                                self.gateway, reason
730                            )));
731                        }
732                    }
733                    _ => {
734                        // Ignore unknown messages in gather state
735                    }
736                }
737            }
738            _ => {}
739        }
740
741        self.results.extend(new_results.clone());
742        Ok(new_results)
743    }
744
745    /// Handle the gateway process closing.
746    pub fn ev_hup(&mut self, rc: i32) {
747        self.last_rc = Some(rc);
748    }
749
750    /// Handle the channel closing.
751    pub fn ev_close(&mut self) {
752        self.state = ChannelState::Closed;
753    }
754
755    /// Get the current channel state.
756    pub fn current_state(&self) -> ChannelState {
757        self.state
758    }
759
760    /// Check if the channel is in gather state (ready for results).
761    pub fn is_gathering(&self) -> bool {
762        self.state == ChannelState::Gather
763    }
764
765    /// Check if the channel is closed.
766    pub fn is_closed(&self) -> bool {
767        self.state == ChannelState::Closed
768    }
