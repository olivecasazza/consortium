3//! Provides TreeWorker, which distributes command execution across a cluster

31/// Remote untar command format for forward tree copy (UNTAR_CMD_FMT).
32///
33/// Single quotes are essential (mirrors ClusterShell); use
34/// [`shell_escape_single_quoted`] on the interpolated path.
35pub const UNTAR_CMD_FMT: &str = "tar -xf - -C '%s'";

37/// Remote tar|base64 command format for tree rcopy (TAR_CMD_FMT).
38///
39/// Args are (srcdir, srcbase). The remote side archives from `srcdir`,
40/// renames the top-level entry to `<name>.<short-hostname>/` and emits
41/// base64 with 64 KiB lines on stdout.
42pub const TAR_CMD_FMT: &str = "tar -cf - -C '%s' \
43--transform \"s,^\\([^/]*\\)[/]*,\\1.$(hostname -s)/,\" \
44'%s' | base64 -w 65536";

46/// Escape a string for inclusion between single quotes in a shell command.
47///
48/// Mirrors the `'`-within-`''` escaping used by ClusterShell's
49/// `_copy_remote`: each `'` becomes `'"'"'`.
50pub fn shell_escape_single_quoted(value: &str) -> String {
51    value.replace('\'', "'\"'\"'")
52}

54/// Build the remote untar command used for forward tree copy.
55pub fn untar_command(dest: &str) -> String {
56    UNTAR_CMD_FMT.replacen("%s", &shell_escape_single_quoted(dest), 1)
57}

59/// Build the remote tar|base64 command used for tree rcopy.
60pub fn tar_command(srcdir: &str, srcbase: &str) -> String {
61    TAR_CMD_FMT
62        .replacen("%s", &shell_escape_single_quoted(srcdir), 1)
63        .replacen("%s", &shell_escape_single_quoted(srcbase), 1)

66/// POSIX-style dirname (upstream Python os.path.dirname semantics for
67/// the inputs ClusterShell produces).
68fn posix_dirname(path: &str) -> &str {
69    let trimmed = path.trim_end_matches('/');
70    if trimmed.is_empty() {
71        // path was only slashes: os.path.dirname('/') == '/'
72        return if path.starts_with('/') { "/" } else { "" };
73    }
74    match trimmed.rfind('/') {
75        Some(0) => "/",
76        Some(pos) => &trimmed[..pos],
77        None => "",
78    }
79}

81/// POSIX-style basename of a normalized path (os.path.basename).
82fn posix_basename(path: &str) -> &str {
83    let trimmed = path.trim_end_matches('/');
84    match trimmed.rfind('/') {
85        Some(pos) => &trimmed[pos + 1..],
86        None => trimmed,
87    }
88}

90/// Minimal os.path.normpath for absolute/relative POSIX paths: collapses
91/// duplicate slashes, resolves `.` and `..` components, strips any
92/// trailing slash. Does not resolve symlinks (like os.path.normpath).
93fn posix_normpath(path: &str) -> String {
94    let absolute = path.starts_with('/');
95    let mut out: Vec<&str> = Vec::new();
96    for comp in path.split('/') {
97        match comp {
98            "" | "." => {}
99            ".." => {
100                if out.last().is_some_and(|last| *last != "..") {
101                    out.pop();
102                } else if !absolute {
103                    out.push("..");
104                }
105            }
106            _ => out.push(comp),
107        }
108    }
109    let joined = out.join("/");
110    match (absolute, joined.is_empty()) {
111        (true, true) => "/".to_string(),
112        (true, false) => format!("/{}", joined),
113        (false, true) => ".".to_string(),
114        (false, false) => joined,
115    }
116}

428    /// Distribute an arbitrary nodeset across next-hop gateways
429    /// (upstream `TreeWorker._distribute`).
430    fn distribute_nodes(&self, nodes: &NodeSet) -> Result<Vec<(NodeSet, NodeSet)>, WorkerError> {
431        let router = self.router.as_ref().ok_or_else(|| {
432            WorkerError::General("router not set; call set_router() or set_topology()".to_string())
433        })?;
434
435        let mut distribution: HashMap<String, NodeSet> = HashMap::new();
436
437        for (gw, dstset) in router.dispatch(nodes) {
438            let gw_str = gw.to_string();
439            distribution
440                .entry(gw_str)
441                .and_modify(|ns| ns.update(&dstset))
442                .or_insert(dstset);
443        }
444
445        Ok(distribution
446            .into_iter()
447            .map(|(k, v)| {
448                let gw = NodeSet::parse(&k).unwrap_or_default();
449                (gw, v)
450            })
451            .collect())
452    }

454    /// The command to run on remote targets through a gateway, if any.
455    ///
456    /// Mirrors upstream `_execute_remote` (command mode) and
457    /// `_copy_remote` (copy/rcopy mode): copy operations send a tar
458    /// pipeline command to the targets instead of a user command.
459    fn remote_shell_command(&self) -> Option<String> {
460        if let Some(ref cmd) = self.command {
461            return Some(cmd.clone());
462        }
463        let source = self.config.source.as_ref()?;
464        if self.config.reverse {
465            // rcopy: archive remotely, transfer base64 tar on stdout
466            let srcdir = posix_dirname(source);
467            let normed = posix_normpath(source);
468            let srcbase = posix_basename(&normed);
469            Some(tar_command(srcdir, srcbase))
470        } else {
471            // forward copy: untar from stdin at the destination
472            let dest = self.config.dest.as_deref().unwrap_or("");
473            Some(untar_command(dest))
474        }
475    }

477    /// Start the tree worker — distribute work across gateways.
478    ///
479    /// This sets up propagation channels for gateway nodes and prepares
480    /// direct execution for leaf nodes.
481    pub fn start(&mut self) -> Result<Vec<(NodeSet, NodeSet)>, WorkerError> {
482        if self.state != WorkerState::Pending {
483            return Err(WorkerError::General(
484                "worker already started or aborted".to_string(),
485            ));
486        }
487
488        self.state = WorkerState::Running;
489
490        let next_hops = self.distribute()?;
491
492        for (gw, targets) in &next_hops {
493            let target_count = targets.len();
494            self.target_count += target_count;
495
496            let gw_str = gw.to_string();
497            let targets_str = targets.to_string();
498
499            // Check if gateway == targets (direct execution, no gateway needed)
500            if gw_str == targets_str {
501                // Direct targets — caller should spawn SSH/local workers for
502                // these; each spawned child worker counts towards _child_count
503                // (upstream TreeWorker._launch).
504                self.child_count += 1;
505                continue;
506            }
507
508            // Gateway-mediated execution: set up propagation channel
509            let channel = PropagationChannel::new(&gw_str);
510            let gw_state = GatewayState {
511                channel,
512                active_targets: targets.iter().collect(),
513                started: false,
514                ctl_unsent_targets: Vec::new(),
515            };
516            self.gateways.insert(gw_str, gw_state);
517        }
518
519        // With no direct children this fires ev_start immediately, like
520        // upstream's `_start()`: `_launch()` then `_check_ini()`.
521        self.check_ini();
522
523        Ok(next_hops)
524    }

526    /// Start a gateway's propagation channel with topology data.
527    ///
528    /// # Arguments
529    /// * `gateway` - The gateway node name.
530    /// * `topology_bytes` - Serialized topology data to send to gateway.
531    pub fn start_gateway(
532        &mut self,
533        gateway: &str,
534        topology_bytes: &[u8],
535    ) -> Result<(), WorkerError> {
536        let command = self.remote_shell_command();
537        let worker_id = self.worker_id;
538        {
539            let gw_state = self
540                .gateways
541                .get_mut(gateway)
542                .ok_or_else(|| WorkerError::General(format!("unknown gateway: {}", gateway)))?;
543
544            if gw_state.started {
545                return Err(WorkerError::General(format!(
546                    "gateway {} already started",
547                    gateway
548                )));
549            }
550
551            gw_state.channel.start(topology_bytes);
552            gw_state.started = true;
553
554            // Send the shell command (command mode) or tar pipeline (copy mode).
555            if let Some(ref cmd) = command {
556                let mut targets: Vec<String> = gw_state.active_targets.iter().cloned().collect();
557                targets.sort();
558                let targets_str = nodeset_from_strings(&targets);
559                gw_state
560                    .channel
561                    .shell(&targets_str, cmd.as_bytes(), worker_id);
562                // ev_pickup is fired by sync_gateway_pickups() once the control
563                // message has actually been sent on the wire (#594).
564                gw_state.ctl_unsent_targets.extend(targets);
565            }
566        }
567
568        self.sync_gateway_pickups(gateway);
569
570        Ok(())
571    }

573    /// Emit pickups for gateway targets whose control message has
574    /// actually been sent on the wire.
575    ///
576    /// Mirrors `PropagationChannel._send_ctl` firing `_emit_pickup`
577    /// after `channel.send` (upstream ClusterShell #594): once channel
578    /// setup is complete, a queued control message has left the local
579    /// process and its targets can no longer be rerouted. Call this
580    /// after the channel processes the configuration ACK (and after
581    /// [`TreeWorker::start_gateway`]); it is a no-op while the control
582    /// message is still queued behind an un-ACKed configuration.
583    pub fn sync_gateway_pickups(&mut self, gateway: &str) {
584        let Some(gw_state) = self.gateways.get_mut(gateway) else {
585            return;
586        };
587        if !gw_state.channel.channel.setup || gw_state.ctl_unsent_targets.is_empty() {
588            return;
589        }
590        let targets = std::mem::take(&mut gw_state.ctl_unsent_targets);
591        for node in targets {
592            self.emit_pickup(&node);
593        }
594    }

596    /// Fire `ev_pickup` for a node, or queue it until `ev_start` fires.
597    ///
598    /// Mirrors upstream `TreeWorker._emit_pickup` (#594): direct child
599    /// workers route pickups through `MetaWorkerEventHandler.ev_pickup`;
600    /// gateway-routed targets route them through the propagation channel
601    /// once the control message leaves the local process (see
602    /// [`TreeWorker::sync_gateway_pickups`]). Emission is skipped once
603    /// the worker has been aborted.
604    pub fn emit_pickup(&mut self, node: &str) {
605        if self.aborted {
606            return;
607        }
608        if self.initialized {
609            self.events.push(TreeWorkerEvent::Pickup {
610                node: node.to_string(),
611            });
612        } else {
613            // ev_start has not fired yet; defer to preserve ordering
614            self.pending_pickups.push(node.to_string());
615        }
616    }

627    /// Fire `ev_start` once all direct child workers have started, then
628    /// flush pickup events buffered during launch.
629    ///
630    /// Mirrors upstream `TreeWorker._check_ini` (#594): this part runs
631    /// once. If the worker is aborted on (or before) `ev_start`, the
632    /// flush stops at the first aborted check and no pickups are
633    /// emitted.
634    pub fn check_ini(&mut self) {
635        if !self.initialized && self.start_count >= self.child_count {
636            self.initialized = true;
637            self.events.push(TreeWorkerEvent::Started);
638            let pending = std::mem::take(&mut self.pending_pickups);
639            for node in pending {
640                if self.aborted {
641                    break;
642                }
643                self.events.push(TreeWorkerEvent::Pickup { node });
644            }
645        }
646    }

657    /// Redistribute and relaunch targets that were running on a failed
658    /// gateway (upstream `TreeWorker._relaunch`).
659    ///
660    /// Emits a `reroute` routing event BEFORE re-dispatching, so a
661    /// rerouted target's `ev_pickup` can only fire afterwards — and,
662    /// because pickup requires the control message to have been sent,
663    /// exactly once (#594). Upstream only reroutes targets whose control
664    /// message never left the failed channel; callers should do the same
665    /// (a target already picked up on the failed gateway must not be
666    /// relaunched).
667    ///
668    /// Returns the new (gateway, targets) distribution; entries with
669    /// gateway == targets are directly reachable and must be handled by
670    /// the caller like any direct target.
671    pub fn relaunch(
672        &mut self,
673        previous_gateway: &str,
674    ) -> Result<Vec<(NodeSet, NodeSet)>, WorkerError> {
675        let gw_state = self.gateways.remove(previous_gateway).ok_or_else(|| {
676            WorkerError::General(format!("unknown gateway: {}", previous_gateway))
677        })?;
678
679        let mut targets: Vec<String> = gw_state.active_targets.into_iter().collect();
680        targets.sort();
681        let target_count = targets.len();
682        // Any control message queued but never sent on the failed channel
683        // dies with it: no pickup was ever emitted for these targets (#594).
684        // (gw_state is dropped here, closing the channel state.)
685
686        self.target_count = self.target_count.saturating_sub(target_count);
687
688        // Routing event fires before re-dispatch (upstream _ev_routing).
689        self.events.push(TreeWorkerEvent::Routing {
690            event: "reroute".to_string(),
691            gateway: previous_gateway.to_string(),
692            targets: targets.join(","),
693        });
694
695        // Re-dispatch the orphaned targets (upstream `_launch(targets)`).
696        let targets_ns = NodeSet::parse(&targets.join(","))
697            .map_err(|e| WorkerError::General(format!("invalid targets: {}", e)))?;
698        let next_hops = self.distribute_nodes(&targets_ns)?;
699
700        for (gw, hop_targets) in &next_hops {
701            self.target_count += hop_targets.len();
702            let gw_str = gw.to_string();
703            if gw_str == hop_targets.to_string() {
704                // became directly reachable — caller spawns a child worker
705                self.child_count += 1;
706                continue;
707            }
708            let new_targets: HashSet<String> = hop_targets.iter().collect();
709            match self.gateways.get_mut(&gw_str) {
710                Some(existing) => {
711                    // Channel to this gateway already exists (it serves other
712                    // targets): send the command for the relaunched targets on
713                    // it, like upstream `_execute_remote` reusing the pchannel.
714                    let mut ts: Vec<String> = new_targets.iter().cloned().collect();
715                    ts.sort();
716                    existing.active_targets.extend(new_targets);
717                    self.shell_on_channel(&gw_str, &ts)?;
718                }
719                None => {
720                    let channel = PropagationChannel::new(&gw_str);
721                    self.gateways.insert(
722                        gw_str,
723                        GatewayState {
724                            channel,
725                            active_targets: new_targets,
726                            started: false,
727                            ctl_unsent_targets: Vec::new(),
728                        },
729                    );
730                }
731            }
732        }
733
734        self.check_ini();
735
736        Ok(next_hops)
737    }

739    /// Send this worker's command to `targets` over an existing gateway
740    /// channel (used when a relaunch lands on an already-started gateway).
741    /// Pickups fire via [`TreeWorker::sync_gateway_pickups`] once the
742    /// control message is actually sent.
743    fn shell_on_channel(&mut self, gateway: &str, targets: &[String]) -> Result<(), WorkerError> {
744        let command = self.remote_shell_command();
745        let worker_id = self.worker_id;
746        {
747            let gw_state = self
748                .gateways
749                .get_mut(gateway)
750                .ok_or_else(|| WorkerError::General(format!("unknown gateway: {}", gateway)))?;
751
752            if let Some(ref cmd) = command {
753                let targets_str = nodeset_from_strings(targets);
754                gw_state
755                    .channel
756                    .shell(&targets_str, cmd.as_bytes(), worker_id);

764    /// Process a propagation result from a gateway.
765    ///
766    /// Call this when the gateway channel receives results.
767    /// Returns true if the worker is now complete.
768    pub fn process_result(
769        &mut self,
770        gateway: &str,
771        result: PropagationResult,
772    ) -> Result<bool, WorkerError> {
773        // Results only arrive after the channel is set up, so any queued
774        // control message has been sent by now (#594).
775        self.sync_gateway_pickups(gateway);
776
777        match result {
778            PropagationResult::StdOut { ref node, ref data } => {
779                self.events.push(TreeWorkerEvent::StdOut {
780                    node: node.clone(),
781                    gateway: gateway.to_string(),
782                    data: data.clone(),
783                });
784            }
785            PropagationResult::StdErr { ref node, ref data } => {
786                self.events.push(TreeWorkerEvent::StdErr {
787                    node: node.clone(),
788                    gateway: gateway.to_string(),
789                    data: data.clone(),
790                });
791            }
792            PropagationResult::Retcode { ref node, rc } => {
793                self.retcodes.insert(node.clone(), rc);
794                self.close_count += 1;
795
796                // Remove from active targets
797                if let Some(gw_state) = self.gateways.get_mut(gateway) {
798                    gw_state.active_targets.remove(node);
799                }
800
801                self.events.push(TreeWorkerEvent::Close {
802                    node: node.clone(),
803                    gateway: gateway.to_string(),
804                    rc,
805                });
806            }
807            PropagationResult::Timeout { ref node } => {
808                self.timeout_nodes.insert(node.clone());
809                self.has_timeout = true;
810                self.close_count += 1;
811
812                if let Some(gw_state) = self.gateways.get_mut(gateway) {
813                    gw_state.active_targets.remove(node);
814                }
815
816                self.events.push(TreeWorkerEvent::Timeout {
817                    node: node.clone(),
818                    gateway: gateway.to_string(),
819                });
820            }
821            PropagationResult::Routing {
822                ref event,
823                ref gateway,
824                ref targets,
825            } => {
826                self.events.push(TreeWorkerEvent::Routing {
827                    event: event.clone(),
828                    gateway: gateway.clone(),
829                    targets: targets.clone(),
830                });
831            }
832        }
833
834        // Check if we're done
835        let done = self.close_count >= self.target_count && self.target_count > 0;
836        if done {
837            self.state = WorkerState::Done;
838        }
839
840        // Check if gateway has no more active targets
841        self.check_gateway_fini(gateway);
842
843        Ok(done)
844    }

910    /// Abort the tree worker.
911    ///
912    /// Marks all remaining gateway targets as failed with EX_PROTOCOL (76).
913    /// Also stops pending and future pickup events (#594): buffered
914    /// pickups are dropped and [`TreeWorker::emit_pickup`] becomes a no-op.
915    pub fn abort(&mut self) {
916        // Stop pending and future pickup events (#594). Upstream sets
917        // _aborted unconditionally, even for an already-finished worker.
918        self.aborted = true;
919        self.pending_pickups.clear();
920
921        if self.is_done() {
922            return;

924

927            let remaining: Vec<String> = gw_state.active_targets.drain().collect();
928            for node in remaining {

930                self.close_count += 1;
931                self.events.push(TreeWorkerEvent::Close {
932                    node,
933                    gateway: gateway.clone(),
934                    rc: 76,
935                });
936            }
937        }
938
939        self.state = WorkerState::Aborted;
940    }

971    /// attributed to `nodes`, as seen from `gateway`.
972    ///
973    /// Mirrors `PropagationChannel._report_stderr` delivering to this

979        data.push(b'\n');

982        // line is stripped as well (\r\n line endings).

987                Some(b'\r') => &line[..line.len() - 1],
988                _ => line,
989            };
990            for node in nodes.iter() {
991                self.events.push(TreeWorkerEvent::StdErr {
992                    node,
993                    gateway: gateway.to_string(),
994                    data: line.to_vec(),
995                });
996            }
997        }

1026    /// gateway (upstream `TreeWorker._on_remote_node_msgline`).
1027    ///
1028    /// For rcopy workers, `stdout` carries the base64-encoded tar stream
1029    /// and is accumulated for later extraction instead of being reported
1030    /// (partial base64 decoding requires a multiple of 4 characters;
1031    /// trailing chars are kept for the next message). Everything else is
1032    /// recorded as a stdout/stderr event.
1033    pub fn on_remote_node_msgline(&mut self, node: &str, msg: &[u8], sname: &str, gateway: &str) {
1034        let is_rcopy_data =
1035            self.config.source.is_some() && self.config.reverse && sname == SNAME_STDOUT;
1036        if !is_rcopy_data {
1037            let event = if sname == SNAME_STDERR {
1038                TreeWorkerEvent::StdErr {
1039                    node: node.to_string(),
1040                    gateway: gateway.to_string(),
1041                    data: msg.to_vec(),
1042                }
1043            } else {
1044                TreeWorkerEvent::StdOut {
1045                    node: node.to_string(),
1046                    gateway: gateway.to_string(),
1047                    data: msg.to_vec(),
1048                }
1049            };
1050            self.events.push(event);
1051            return;
1052        }
1053
1054        // rcopy only: we expect base64 encoded tar content on stdout
1055        let state = self.rcopy_states.entry(node.to_string()).or_default();
1056        state.encoded_tail.extend_from_slice(msg);
1057        // partial base64 decoding requires a multiple of 4 characters
1058        let decodable = state.encoded_tail.len() / 4 * 4;
1059        let chunk: Vec<u8> = state.encoded_tail.drain(..decodable).collect();
1060        if let Ok(text) = std::str::from_utf8(&chunk) {

1063            if let Ok(decoded) = crate::communication::base64_decode(text) {
1064                state.tar_bytes.extend_from_slice(&decoded);
1065            }
1066            // Undecodable chunks are dropped here; the resulting corrupt
1067            // tar stream is reported at finalize time (upstream 973870b).
1068        }

1071    /// Finalize rcopy for a node whose remote side has closed: extract
1072    /// the accumulated tar stream into the destination directory.
1073    ///
1074    /// Mirrors the reverse-copy branch of upstream
1075    /// `TreeWorker._on_remote_node_close`. Extraction uses
1076    /// [`TarExtractFilter::FullyTrusted`] semantics (PEP 706, upstream
1077    /// d2fcd16). Any failure — corrupt archive, I/O error — is reported
1078    /// as a stderr event attributed to the node carrying the gateway's
1079    /// message (upstream 973870b), never panicked or silently dropped.
1080    ///
1081    /// Returns the number of extracted bytes on success; errors are
1082    /// reported through events, so this only fails on worker misuse
1083    /// (missing destination).
1084    pub fn finalize_rcopy_node(&mut self, node: &str, gateway: &str) -> Result<usize, WorkerError> {
1085        if !(self.config.source.is_some() && self.config.reverse) {
1086            return Ok(0);
1087        }
1088        let Some(state) = self.rcopy_states.remove(node) else {
1089            // no rcopy buffer received from this node
1090            return Ok(0);
1091        };
1092
1093        let dest =
1094            self.config.dest.clone().ok_or_else(|| {
1095                WorkerError::General("missing dest parameter for rcopy".to_string())
1096            })?;
1097
1098        // Upstream flushes the raw leftover base64 tail into the archive
1099        // before extracting; mirror that (it is empty for well-formed
1100        // streams).
1101        let mut tar_bytes = state.tar_bytes;
1102        tar_bytes.extend_from_slice(&state.encoded_tail);
1103
1104        match extract_tar(
1105            std::path::Path::new(&dest),
1106            &tar_bytes,
1107            TarExtractFilter::FullyTrusted,
1108        ) {
1109            Ok(written) => Ok(written),
1110            Err(err) => {
1111                // Report the extraction error as stderr bytes (as bytes,
1112                // like any other message — upstream 973870b).
1113                self.events.push(TreeWorkerEvent::StdErr {
1114                    node: node.to_string(),
1115                    gateway: gateway.to_string(),
1116                    data: err.message.into_bytes(),
1117                });
1118                Ok(0)
1119            }
1120        }
1121    }

1206/// Size of a tar header/data block.
1207const TAR_BLOCK: usize = 512;
1208
1209/// Parsed header fields needed for extraction.
1210struct TarHeader {
1211    /// Member path (prefix + name, GNU longname or pax path applied).
1212    name: String,
1213    /// Permission bits (octal mode field).
1214    mode: u32,
1215    /// Typeflag byte.
1216    typeflag: u8,
1217    /// Link target for symlink/hardlink members.
1218    linkname: String,
1219}
1220
1221/// Parse an octal-or-base256 numeric tar header field.
1222fn parse_tar_number(field: &[u8]) -> Option<u64> {
1223    if field.is_empty() {
1224        return None;
1225    }
1226    // base-256 (GNU large values): high bit of first byte set
1227    if field[0] & 0x80 != 0 {
1228        let mut value: u64 = (field[0] & 0x7F) as u64;
1229        for byte in &field[1..] {
1230            value = value.checked_shl(8)? | u64::from(*byte);
1231        }
1232        return Some(value);
1233    }
1234    // octal ASCII, terminated by NUL/space, possibly space-padded
1235    let text: String = field
1236        .iter()
1237        .take_while(|b| **b != 0 && **b != b' ')
1238        .map(|b| *b as char)
1239        .collect();
1240    if text.is_empty() {
1241        return Some(0);
1242    }
1243    u64::from_str_radix(&text, 8).ok()
1244}

1246/// Extract a ustar archive into `dest` with explicit trust semantics.
1247///
1248/// Returns the total number of payload bytes written. Failures mirror
1249/// the exception kinds upstream reports as node stderr (commit 973870b):
1250/// corrupt archives (Python `tarfile.TarError`) and I/O failures
1251/// (`IOError`/`OSError`) both surface as [`TarExtractError`].
1252///
1253/// With [`TarExtractFilter::FullyTrusted`], stored permission bits are
1254/// applied verbatim (`mode & 0o7777`, ignoring umask) and member paths
1255/// are used unsanitized — matching PEP 706 `fully_trusted`, which
1256/// preserves the pre-3.12 behavior for trusted ClusterShell gateways.
1257pub fn extract_tar(
1258    dest: &std::path::Path,
1259    bytes: &[u8],
1260    _filter: TarExtractFilter,
1261) -> Result<usize, TarExtractError> {
1262    use std::io::Write;
1263    use std::os::unix::fs::PermissionsExt;
1264
1265    if bytes.is_empty() {
1266        // tarfile.ReadError("empty file") on a zero-byte archive
1267        return Err(TarExtractError::corrupt("empty file"));
1268    }
1269
1270    let mut written = 0usize;
1271    let mut offset = 0usize;
1272    // GNU longname/longlink carried over to the next header
1273    let mut pending_name: Option<String> = None;
1274    let mut pending_linkname: Option<String> = None;
1275    // directory modes are applied after all members are extracted, like
1276    // tarfile delaying directory chmod (dirs may be read-only)
1277    let mut dir_modes: Vec<(std::path::PathBuf, u32)> = Vec::new();
1278
1279    let mut terminated = false;
1280    while offset + TAR_BLOCK <= bytes.len() {
1281        let block = &bytes[offset..offset + TAR_BLOCK];
1282
1283        // a zero block terminates the archive
1284        if block.iter().all(|b| *b == 0) {
1285            terminated = true;
1286            break;
1287        }
1288
1289        // verify header checksum: chksum field counts as 8 spaces
1290        let stored = parse_tar_number(&block[148..156])
1291            .ok_or_else(|| TarExtractError::corrupt("bad checksum field"))?;
1292        let computed: u64 = block[..148]
1293            .iter()
1294            .chain(std::iter::repeat(&b' ').take(8))
1295            .chain(block[156..].iter())
1296            .map(|b| u64::from(*b))
1297            .sum();
1298        if stored != computed {
1299            return Err(TarExtractError::corrupt("bad checksum"));
1300        }
1301
1302        let mut name_bytes: Vec<u8> = block[0..100]
1303            .iter()
1304            .take_while(|b| **b != 0)
1305            .cloned()
1306            .collect();
1307        // ustar prefix (ignored for old v7 headers without magic)
1308        if &block[257..262] == b"ustar" {
1309            let prefix: Vec<u8> = block[345..500]
1310                .iter()
1311                .take_while(|b| **b != 0)
1312                .cloned()
1313                .collect();
1314            if !prefix.is_empty() {
1315                let mut full = prefix;
1316                full.push(b'/');
1317                full.extend_from_slice(&name_bytes);
1318                name_bytes = full;
1319            }
1320        }
1321        let name = String::from_utf8_lossy(&name_bytes).into_owned();
1322        let mode = parse_tar_number(&block[100..108]).unwrap_or(0) as u32;
1323        let size = parse_tar_number(&block[124..136])
1324            .ok_or_else(|| TarExtractError::corrupt("bad size field"))?;
1325        let typeflag = block[156];
1326        let linkname = String::from_utf8_lossy(
1327            &block[157..257]
1328                .iter()
1329                .take_while(|b| **b != 0)
1330                .cloned()
1331                .collect::<Vec<u8>>(),
1332        )
1333        .into_owned();
1334
1335        offset += TAR_BLOCK;
1336        let data_blocks = (size as usize).div_ceil(TAR_BLOCK);
1337        let data_len = data_blocks * TAR_BLOCK;
1338        if offset + data_len > bytes.len() {
1339            return Err(TarExtractError::corrupt("truncated tar archive"));
1340        }
1341        let payload = &bytes[offset..offset + size as usize];
1342        offset += data_len;
1343
1344        // GNU/pax name overrides apply to the next real header
1345        match typeflag {
1346            b'L' => {
1347                pending_name = Some(
1348                    String::from_utf8_lossy(payload)
1349                        .trim_end_matches('\0')
1350                        .to_string(),
1351                );
1352                continue;
1353            }
1354            b'K' => {
1355                pending_linkname = Some(
1356                    String::from_utf8_lossy(payload)
1357                        .trim_end_matches('\0')
1358                        .to_string(),
1359                );
1360                continue;
1361            }
1362            b'x' | b'g' => {
1363                // pax extended header: honor path=/linkpath= records
1364                for (key, value) in parse_pax_records(payload) {
1365                    if typeflag == b'x' {
1366                        match key.as_str() {
1367                            "path" => pending_name = Some(value),
1368                            "linkpath" => pending_linkname = Some(value),
1369                            _ => {}
1370                        }
1371                    }
1372                }
1373                continue;
1374            }
1375            _ => {}
1376        }
1377
1378        let header = TarHeader {
1379            name: pending_name.take().unwrap_or(name),
1380            mode,
1381            typeflag,
1382            linkname: pending_linkname.take().unwrap_or(linkname),
1383        };
1384
1385        match header.typeflag {
1386            // regular file (also '\0' and contiguous '7')
1387            b'0' | 0 | b'7' => {
1388                let path = dest.join(&header.name);
1389                if let Some(parent) = path.parent() {
1390                    std::fs::create_dir_all(parent)?;
1391                }
1392                let mut file = std::fs::File::create(&path)?;
1393                file.write_all(payload)?;
1394                // fully_trusted: honor stored mode bits verbatim
1395                file.set_permissions(std::fs::Permissions::from_mode(header.mode & 0o7777))?;
1396                written += payload.len();
1397            }
1398            b'5' => {
1399                let path = dest.join(&header.name);
1400                std::fs::create_dir_all(&path)?;
1401                dir_modes.push((path, header.mode));
1402            }
1403            b'2' => {
1404                let path = dest.join(&header.name);
1405                if let Some(parent) = path.parent() {
1406                    std::fs::create_dir_all(parent)?;
1407                }
1408                std::os::unix::fs::symlink(&header.linkname, &path)?;
1409            }
1410            b'1' => {
1411                let path = dest.join(&header.name);
1412                if let Some(parent) = path.parent() {
1413                    std::fs::create_dir_all(parent)?;
1414                }
1415                std::fs::hard_link(dest.join(&header.linkname), &path)?;
1416            }
1417            other => {
1418                return Err(TarExtractError::corrupt(&format!(
1419                    "unsupported tar member type '{}' for '{}'",
1420                    other as char, header.name
1421                )));
1422            }
1423        }
1424    }
1425
1426    // Without a terminating zero block, non-zero trailing bytes mean a
1427    // truncated final header (tarfile.ReadError). Content AFTER a zero
1428    // block is ignored like tarfile does — this is what makes upstream's
1429    // raw leftover-tail flush at finalize time harmless (#594-era quirk
1430    // preserved in TreeWorker::finalize_rcopy_node).
1431    if !terminated && bytes[offset..].iter().any(|b| *b != 0) {
1432        return Err(TarExtractError::corrupt("truncated tar header"));
1433    }
1434
1435    // apply delayed directory modes (deepest first)
1436    dir_modes.sort_by_key(|(path, _)| std::cmp::Reverse(path.components().count()));
1437    for (path, mode) in dir_modes {
1438        std::fs::set_permissions(&path, std::fs::Permissions::from_mode(mode & 0o7777))?;
1439    }
1440
1441    Ok(written)
1442}

1444/// Parse pax extended-header records (`LEN KEY=VALUE\n`).
1445fn parse_pax_records(payload: &[u8]) -> Vec<(String, String)> {
1446    let mut records = Vec::new();
1447    let text = String::from_utf8_lossy(payload);
1448    let mut rest = text.as_ref();
1449    while !rest.is_empty() {
1450        // record length is the decimal prefix up to the first space
1451        let Some(space) = rest.find(' ') else { break };
1452        let Ok(len) = rest[..space].parse::<usize>() else {
1453            break;
1454        };
1455        if len == 0 || len > rest.len() {
1456            break;
1457        }
1458        let record = &rest[space + 1..len];
1459        if let Some(eq) = record.find('=') {
1460            let key = record[..eq].to_string();
1461            let value = record[eq + 1..].trim_end_matches('\n').to_string();
1462            records.push((key, value));
1463        }
1464        rest = &rest[len..];
1465    }
1466    records
1467}

1761    /// Build a ustar header block with a correct checksum.
1762    fn tar_header_block(name: &str, mode: u32, size: usize, typeflag: u8) -> Vec<u8> {
1763        assert!(name.len() <= 100);
1764        let mut h = vec![0u8; TAR_BLOCK];
1765        h[..name.len()].copy_from_slice(name.as_bytes());
1766        let mode_s = format!("{:o}", mode);
1767        h[100..100 + mode_s.len()].copy_from_slice(mode_s.as_bytes());
1768        let size_s = format!("{:o}", size);
1769        h[124..124 + size_s.len()].copy_from_slice(size_s.as_bytes());
1770        h[156] = typeflag;
1771        h[257..262].copy_from_slice(b"ustar");
1772        for b in h.iter_mut().take(156).skip(148) {
1773            *b = b' ';
1774        }
1775        let sum: u64 = h.iter().map(|b| u64::from(*b)).sum();
1776        let chk = format!("{:o}\0 ", sum);
1777        h[148..148 + chk.len()].copy_from_slice(chk.as_bytes());
1778        h
1779    }

1781    /// Build a minimal ustar archive from directory and file members.
1782    fn make_tar(dirs: &[(&str, u32)], files: &[(&str, u32, &[u8])]) -> Vec<u8> {
1783        let mut out = Vec::new();
1784        for (name, mode) in dirs {
1785            out.extend(tar_header_block(name, *mode, 0, b'5'));
1786        }
1787        for (name, mode, data) in files {
1788            out.extend(tar_header_block(name, *mode, data.len(), b'0'));
1789            out.extend_from_slice(data);
1790            let pad = (TAR_BLOCK - data.len() % TAR_BLOCK) % TAR_BLOCK;
1791            out.extend(std::iter::repeat(0u8).take(pad));
1792        }
1793        // two terminating zero blocks
1794        out.extend(std::iter::repeat(0u8).take(2 * TAR_BLOCK));
1795        out
1796    }
